export function isValidAnchor(text: string): boolean {
  const words = text.trim().split(/\s+/);
  if (words.length < 4) return false;
  // Simple regex to check if it's just "City, ST"
  const isBareGeo = /^[a-zA-Z\s]+,?\s?[A-Z]{2}$/.test(text.trim());
  return !isBareGeo;
}
