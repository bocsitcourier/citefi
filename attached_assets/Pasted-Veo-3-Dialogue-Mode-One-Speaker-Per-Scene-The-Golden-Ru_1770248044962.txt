Veo 3 Dialogue Mode: One Speaker Per Scene
The Golden Rule

Each scene segment (7-8 seconds) must have exactly ONE speaker. Never alternate speakers within a scene. Speaker changes require a new scene/extension.


Why This Matters
Veo 3's lip-sync works by:

Detecting ONE face/character as the speaker
Animating THAT character's mouth to match the dialogue
If multiple characters speak in one scene, lip-sync breaks or becomes unpredictable

Updated Scene Interface
typescript// scene-types.ts

interface SceneSegment {
  id: number;
  duration: number;          // 6-8 seconds per scene
  
  // Visual
  visualPrompt: string;      // What we see
  cameraFocus: string;       // WHO the camera is focused on (the speaker!)
  
  // Audio - SINGLE SPEAKER ONLY
  speaker: string;           // Exactly ONE speaker name
  dialogue: string;          // What THIS speaker says (no alternation)
  voiceDirection: string;    // How they say it
  
  // Technical
  negativePrompt?: string;
  transitionFrom?: string;   // How we got here from previous scene
}

// Validation function
function validateScene(scene: SceneSegment): void {
  // Check for multiple speaker indicators
  const multiSpeakerPatterns = [
    /\w+:\s*"/g,              // "Name: " dialogue format
    /"\s*\n\s*"/g,            // Multiple quoted sections
    /said \w+.*said \w+/gi,   // Multiple "said X" attributions
  ];
  
  for (const pattern of multiSpeakerPatterns) {
    if (pattern.test(scene.dialogue)) {
      throw new Error(
        `Scene ${scene.id} appears to have multiple speakers. ` +
        `Each scene must have exactly ONE speaker. ` +
        `Split into separate scenes for dialogue exchanges.`
      );
    }
  }
  
  // Check dialogue doesn't contain speaker labels
  if (scene.dialogue.includes(':') && /^[A-Z][a-z]+:/.test(scene.dialogue)) {
    throw new Error(
      `Scene ${scene.id}: Don't include speaker labels in dialogue. ` +
      `Use the 'speaker' field instead. ` +
      `Dialogue should be pure speech: "${scene.dialogue}"`
    );
  }
}

function validateScript(scenes: SceneSegment[]): void {
  scenes.forEach(validateScene);
  
  // Check total duration
  const totalDuration = scenes.reduce((sum, s) => sum + s.duration, 0);
  if (totalDuration > 148) {
    throw new Error(`Total duration ${totalDuration}s exceeds Veo 3.1 limit of 148s`);
  }
  
  console.log(`✅ Script valid: ${scenes.length} scenes, ${totalDuration}s total`);
}

Correct Prompt Format for Veo 3
typescript// veo-prompt-builder.ts

function buildVeoPrompt(scene: SceneSegment): string {
  // Validate single speaker
  if (!scene.speaker) {
    throw new Error(`Scene ${scene.id} missing required 'speaker' field`);
  }
  
  return `
${scene.visualPrompt}

SPEAKER: ${scene.speaker}
CAMERA FOCUS: The camera is on ${scene.speaker}, who is speaking.

${scene.speaker} says: "${scene.dialogue}"

Voice delivery: ${scene.voiceDirection}

IMPORTANT: Only ${scene.speaker} speaks in this scene. 
Other characters present are silent and listening.
Animate lip-sync ONLY for ${scene.speaker}.
  `.trim();
}

function buildNegativePrompt(scene: SceneSegment, otherCharacters: string[]): string {
  const silentCharacters = otherCharacters
    .filter(c => c !== scene.speaker)
    .map(c => `${c} speaking`)
    .join(', ');
  
  return `
(no subtitles, no captions, no text overlay)
(no background music unless specified)
(no multiple people talking)
(no ${silentCharacters})
(no dialogue from anyone except ${scene.speaker})
(no mouth movement on non-speaking characters)
  `.trim();
}

KEY RULES:
✅ One speaker per box (scene)
✅ Camera focuses on the speaker
✅ Speaker change = New scene = New extension

❌ Never two speakers in same box
❌ Never "dialogue ping-pong" within a scene

Chained Extension with Speaker Tracking
typescript// veo-extension-with-speakers.ts

class VeoDialogueExtension {
  private currentSpeaker: string | null = null;
  
  async extendScene(
    previousClip: GeneratedClip,
    scene: SceneSegment
  ): Promise<GeneratedClip> {
    
    // Track speaker changes for voice consistency
    const speakerChanged = this.currentSpeaker !== scene.speaker;
    
    const prompt = this.buildPrompt(scene, speakerChanged);
    
    console.log(`  Scene ${scene.id}: ${scene.speaker} speaks`);
    if (speakerChanged) {
      console.log(`    ↳ Speaker changed from ${this.currentSpeaker || 'none'}`);
    }
    
    const result = await this.callVeoAPI({
      prompt: prompt,
      negativePrompt: this.buildNegativePrompt(scene),
      seedFrame: previousClip.endFrame,
      audioSeed: speakerChanged ? null : previousClip.audioSeed, // Reset audio seed on speaker change
      duration: scene.duration
    });
    
    this.currentSpeaker = scene.speaker;
    return result;
  }
  
  private buildPrompt(scene: SceneSegment, speakerChanged: boolean): string {
    return `
Continue from previous frame.

${scene.visualPrompt}

=== SPEAKER INFORMATION ===
CURRENT SPEAKER: ${scene.speaker} (ONLY this character speaks)
${speakerChanged ? 'NOTE: This is a NEW speaker. Previous speaker is now SILENT.' : 'Same speaker continues.'}

${scene.speaker} says: "${scene.dialogue}"
Delivery: ${scene.voiceDirection}

=== LIP-SYNC RULES ===
- Animate mouth ONLY for ${scene.speaker}
- All other characters have CLOSED mouths
- Camera focuses on ${scene.speaker}

=== VOICE CONSISTENCY ===
${speakerChanged 
  ? `Establish ${scene.speaker}'s voice. Different from previous speaker.`
  : `Maintain exact same voice as previous scene for ${scene.speaker}.`
}
    `.trim();
  }
  
  private buildNegativePrompt(scene: SceneSegment): string {
    return `
(no subtitles, no captions)
(no multiple speakers)
(no one speaking except ${scene.speaker})
(no background characters talking)
(no lip movement on silent characters)
(no voice change mid-scene)
    `.trim();
  }
}
