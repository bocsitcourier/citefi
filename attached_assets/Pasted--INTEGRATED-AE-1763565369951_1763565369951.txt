/**
 * ============================================================================
 * INTEGRATED AEO CONTENT FACTORY WITH RESEARCH
 * ============================================================================
 * 
 * Complete workflow:
 * 1. Research experts in the topic
 * 2. Scrape Reddit for real user questions
 * 3. Analyze search intent and patterns
 * 4. Build data-driven prompt
 * 5. Generate article with Gemini
 * 6. Add intelligent links with GPT-4
 * 7. Generate social/podcast/video
 * 
 * npm install axios cheerio @google/generative-ai openai dotenv
 */

const { GoogleGenerativeAI } = require('@google/generative-ai');
const OpenAI = require('openai');
const { ExpertDiscovery, RedditResearch, DynamicPromptBuilder } = require('./expert-reddit-research');
require('dotenv').config();

// ============================================================================
// ENHANCED CONTENT FACTORY WITH RESEARCH
// ============================================================================

class ResearchDrivenContentFactory {
  constructor(config = {}) {
    // Initialize AI clients
    this.gemini = new GoogleGenerativeAI(config.geminiApiKey || process.env.GEMINI_API_KEY);
    this.gpt4 = new OpenAI({ apiKey: config.openaiApiKey || process.env.OPENAI_API_KEY });
    
    // Initialize research modules
    this.expertDiscovery = new ExpertDiscovery({
      braveApiKey: config.braveApiKey || process.env.BRAVE_API_KEY
    });
    this.redditResearch = new RedditResearch({
      redditClientId: config.redditClientId,
      redditSecret: config.redditSecret
    });
    this.promptBuilder = new DynamicPromptBuilder();

    // Configuration
    this.brandData = config.brandData || {};
    this.geminiModel = this.gemini.getGenerativeModel({ model: 'gemini-2.0-flash-exp' });

    console.log('✅ Research-Driven Content Factory initialized');
  }

  // ==========================================================================
  // MAIN WORKFLOW
  // ==========================================================================

  /**
   * Complete research-driven content generation
   */
  async generateWithResearch(topic, location, keyword, options = {}) {
    console.log(`\n${'='.repeat(70)}`);
    console.log(`🚀 RESEARCH-DRIVEN GENERATION`);
    console.log(`${'='.repeat(70)}`);
    console.log(`Topic: ${topic}`);
    console.log(`Location: ${location}`);
    console.log(`Keyword: ${keyword}`);
    console.log(`${'='.repeat(70)}\n`);

    const startTime = Date.now();

    try {
      // PHASE 1: Research
      console.log('📚 PHASE 1: Research & Discovery');
      const research = await this.conductResearch(topic, keyword, location, options);

      // PHASE 2: Build Dynamic Prompt
      console.log('\n📝 PHASE 2: Building Data-Driven Prompt');
      const promptData = await this.buildEnhancedPrompt(
        topic,
        location,
        keyword,
        research,
        options
      );

      // PHASE 3: Generate Article
      console.log('\n✍️  PHASE 3: Generating Article with Gemini');
      const article = await this.generateArticleWithPrompt(promptData.prompt);

      // PHASE 4: Add Intelligent Links
      if (options.internalPages && options.internalPages.length > 0) {
        console.log('\n🔗 PHASE 4: Adding Intelligent Hyperlinks with GPT-4');
        article.html = await this.addLinks(article.html, options.internalPages);
      }

      // PHASE 5: Generate Derivative Content
      console.log('\n📱 PHASE 5: Generating Multi-Format Content');
      const [socialPosts, podcastScript, videoScript] = await Promise.all([
        this.generateSocialMedia(article, topic, location, research),
        this.generatePodcastScript(article, topic, location, research),
        this.generateVideoScript(article, topic, location, research)
      ]);

      const duration = ((Date.now() - startTime) / 1000).toFixed(2);
      
      console.log(`\n${'='.repeat(70)}`);
      console.log(`✅ GENERATION COMPLETE`);
      console.log(`${'='.repeat(70)}`);
      console.log(`Duration: ${duration}s`);
      console.log(`Article: ${article.wordCount} words`);
      console.log(`Research: ${research.experts.length} experts, ${research.redditQuestions.length} questions`);
      console.log(`${'='.repeat(70)}\n`);

      return {
        article,
        social: socialPosts,
        podcast: podcastScript,
        video: videoScript,
        research,
        metadata: {
          topic,
          location,
          keyword,
          generatedAt: new Date().toISOString(),
          generationTime: duration + 's',
          researchQuality: this.assessResearchQuality(research)
        }
      };

    } catch (error) {
      console.error('❌ Generation failed:', error);
      throw error;
    }
  }

  // ==========================================================================
  // RESEARCH PHASE
  // ==========================================================================

  async conductResearch(topic, keyword, location, options = {}) {
    const research = {};

    // Run expert discovery and Reddit research in parallel
    const [experts, redditQuestions] = await Promise.all([
      this.researchExperts(keyword, options),
      this.researchRedditQuestions(keyword, options)
    ]);

    research.experts = experts;
    research.redditQuestions = redditQuestions;

    // Analyze the research
    research.analysis = this.analyzeResearch(experts, redditQuestions);

    return research;
  }

  async researchExperts(keyword, options) {
    try {
      const experts = await this.expertDiscovery.findExperts(
        keyword,
        options.industry || 'general',
        { limit: 10 }
      );

      console.log(`   ✓ Found ${experts.length} experts`);
      return experts;

    } catch (error) {
      console.error('   ✗ Expert discovery failed:', error.message);
      return [];
    }
  }

  async researchRedditQuestions(keyword, options) {
    try {
      const questions = await this.redditResearch.findQuestions(
        keyword,
        {
          subreddits: options.subreddits,
          limit: 30
        }
      );

      console.log(`   ✓ Found ${questions.length} Reddit questions`);
      
      // Group by intent
      const byIntent = {};
      questions.forEach(q => {
        byIntent[q.intent] = byIntent[q.intent] || [];
        byIntent[q.intent].push(q);
      });

      console.log(`   ✓ Intent distribution:`, 
        Object.entries(byIntent).map(([intent, qs]) => 
          `${intent}: ${qs.length}`
        ).join(', ')
      );

      return questions;

    } catch (error) {
      console.error('   ✗ Reddit research failed:', error.message);
      return [];
    }
  }

  analyzeResearch(experts, questions) {
    const analysis = {
      expertiseLevel: this.calculateExpertiseLevel(experts),
      primaryIntent: this.getPrimaryIntent(questions),
      questionThemes: this.extractQuestionThemes(questions),
      contentGaps: this.identifyContentGaps(questions),
      targetAudience: this.identifyTargetAudience(questions)
    };

    return analysis;
  }

  calculateExpertiseLevel(experts) {
    if (experts.length === 0) return 'low';
    
    const avgCredibility = experts.reduce((sum, e) => sum + e.credibilityScore, 0) / experts.length;
    
    if (avgCredibility > 0.8) return 'high';
    if (avgCredibility > 0.6) return 'medium';
    return 'low';
  }

  getPrimaryIntent(questions) {
    const intents = {};
    questions.forEach(q => {
      intents[q.intent] = (intents[q.intent] || 0) + 1;
    });

    return Object.entries(intents)
      .sort((a, b) => b[1] - a[1])[0]?.[0] || 'general';
  }

  extractQuestionThemes(questions) {
    const allKeywords = questions.flatMap(q => q.keywords);
    const keywordFreq = {};
    
    allKeywords.forEach(kw => {
      keywordFreq[kw] = (keywordFreq[kw] || 0) + 1;
    });

    return Object.entries(keywordFreq)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 10)
      .map(([keyword, count]) => ({ keyword, count }));
  }

  identifyContentGaps(questions) {
    const gaps = [];
    
    // Find frequently asked questions with low answer quality
    const highPriorityQuestions = questions
      .filter(q => q.priority > 20)
      .slice(0, 5);

    gaps.push(...highPriorityQuestions.map(q => ({
      question: q.title,
      priority: q.priority,
      subreddit: q.subreddit
    })));

    return gaps;
  }

  identifyTargetAudience(questions) {
    const subreddits = [...new Set(questions.map(q => q.subreddit))];
    
    // Infer audience from subreddits
    const audienceMap = {
      'smallbusiness': 'Small business owners',
      'Entrepreneur': 'Entrepreneurs and startups',
      'SEO': 'SEO professionals',
      'marketing': 'Marketing professionals',
      'learnprogramming': 'Beginner developers',
      'webdev': 'Web developers',
      'personalfinance': 'Individual investors'
    };

    const audiences = subreddits
      .map(sr => audienceMap[sr])
      .filter(Boolean);

    return audiences.length > 0 ? audiences[0] : 'General audience';
  }

  // ==========================================================================
  // PROMPT BUILDING
  // ==========================================================================

  async buildEnhancedPrompt(topic, location, keyword, research, options) {
    const { experts, redditQuestions, analysis } = research;

    const prompt = `You are an expert content strategist creating a comprehensive article based on REAL user research.

TOPIC: ${topic}
LOCATION: ${location}
PRIMARY KEYWORD: ${keyword}
BRAND: ${this.brandData.name || 'Your Company'}

═══════════════════════════════════════════════════════════════
RESEARCH INSIGHTS
═══════════════════════════════════════════════════════════════

EXPERTISE LEVEL: ${analysis.expertiseLevel}
PRIMARY USER INTENT: ${analysis.primaryIntent}
TARGET AUDIENCE: ${analysis.targetAudience}

KEY THEMES FROM RESEARCH:
${analysis.questionThemes.map((t, i) => `${i + 1}. ${t.keyword} (mentioned ${t.count} times)`).join('\n')}

═══════════════════════════════════════════════════════════════
TOP 10 REAL QUESTIONS FROM REDDIT
═══════════════════════════════════════════════════════════════

These are ACTUAL questions people are asking right now. Your article MUST answer all of these:

${redditQuestions.slice(0, 10).map((q, i) => 
`${i + 1}. "${q.title}"
   - Source: r/${q.subreddit}
   - Engagement: ${q.score} upvotes, ${q.numComments} comments
   - Intent: ${q.intent}
   - Priority Score: ${Math.round(q.priority)}`
).join('\n\n')}

ADDITIONAL QUESTIONS TO ADDRESS:
${redditQuestions.slice(10, 20).map(q => `• ${q.title}`).join('\n')}

═══════════════════════════════════════════════════════════════
EXPERT PERSPECTIVES TO REFERENCE
═══════════════════════════════════════════════════════════════

${experts.slice(0, 5).map((expert, i) => 
`${i + 1}. ${expert.name} - ${expert.title}
   Credibility: ${(expert.credibility * 100).toFixed(0)}%
   Expertise: ${expert.topic}
   Key Insight: "${expert.snippet}"
   Source: ${expert.source}`
).join('\n\n')}

═══════════════════════════════════════════════════════════════
CONTENT GAPS TO FILL
═══════════════════════════════════════════════════════════════

Based on Reddit analysis, users are struggling with:
${analysis.contentGaps.map((gap, i) => 
`${i + 1}. ${gap.question} (Priority: ${Math.round(gap.priority)})`
).join('\n')}

═══════════════════════════════════════════════════════════════
ARTICLE STRUCTURE (OPTIMIZED FOR ${analysis.primaryIntent} INTENT)
═══════════════════════════════════════════════════════════════

1. ANSWER-FIRST PARAGRAPH (200 words):
   - Directly answer the #1 Reddit question: "${redditQuestions[0]?.title}"
   - Include specific data or statistics
   - Make it quotable for AI engines
   - Add ${location} context in first paragraph

2. TABLE OF CONTENTS:
   Base section titles on the top Reddit questions

3. MAIN SECTIONS (7-9 sections):
   
   ${this.generateSectionOutline(redditQuestions, analysis.primaryIntent)}

4. EXPERT ROUNDUP SECTION:
   "What the Experts Say About ${keyword}"
   - Quote or paraphrase the top 3-5 experts found
   - Show different perspectives
   - Add credibility to your content

5. COMPARISON TABLE (if relevant to ${analysis.primaryIntent}):
   ${analysis.primaryIntent === 'comparison' ? 
     'Create detailed comparison of options mentioned in Reddit discussions' : 
     'Create a comparison of approaches or methods if relevant'}

6. COMMON MISTAKES SECTION:
   "Mistakes to Avoid (Based on Reddit Discussions)"
   - Extract common mistakes people mention on Reddit
   - Provide solutions

7. REDDIT-INSPIRED FAQ (8-10 questions):
   Use actual Reddit questions, reworded for SEO:
   ${redditQuestions.slice(0, 8).map(q => `- ${q.title}`).join('\n   ')}

8. LOCAL CONTEXT SECTION:
   "${topic} in ${location}: What You Need to Know"
   - ${location}-specific considerations
   - Local examples or case studies
   - Regional differences or regulations

9. CONCLUSION & NEXT STEPS:
   - Summarize key takeaways
   - Provide actionable next steps
   - Address "what should I do now?" from Reddit perspective

═══════════════════════════════════════════════════════════════
CRITICAL WRITING REQUIREMENTS
═══════════════════════════════════════════════════════════════

✓ CONVERSATIONAL TONE: Write like you're answering a Reddit question
✓ PRACTICAL ADVICE: Every section must have actionable takeaways
✓ DATA-DRIVEN: Include specific numbers, not vague claims
✓ ACKNOWLEDGE DEBATE: If Reddit shows conflicting opinions, address both sides
✓ REAL EXAMPLES: Use scenarios similar to what Reddit users describe
✓ NO FLUFF: Get straight to the point (Reddit users hate fluff)

E-E-A-T OPTIMIZATION:
- Experience: Reference the Reddit discussions as real user experiences
- Expertise: Cite the experts discovered in research (by name)
- Authoritativeness: Use credible data sources
- Trustworthiness: Be transparent about limitations and uncertainties

PASSAGE-LEVEL OPTIMIZATION:
- Every paragraph: 3-5 sentences, self-contained
- Start each paragraph with the main point
- Each paragraph should be extractable by AI

LOCAL OPTIMIZATION (${location}):
- Mention ${location} in first 3 paragraphs
- Include ${location}-specific data throughout
- Add local examples in at least 3 sections

WORD COUNT: 2200-2800 words

FORMATTING:
- Use semantic HTML: <article>, <h1>, <h2>, <h3>, <p>, <ul>, <ol>, <table>
- No hyperlinks (handled separately)
- Include "Last Updated: ${new Date().toLocaleDateString()}"
- Add author bio at end

AUTHOR BIO:
"This article was written by ${this.brandData.expertName || 'Expert Name'}, ${this.brandData.expertTitle || 'Senior Strategist'} at ${this.brandData.name || 'Your Company'}, based on extensive research including analysis of real user questions from Reddit communities and insights from industry experts. Last updated: ${new Date().toLocaleDateString()}."

═══════════════════════════════════════════════════════════════

Generate the complete article now, ensuring EVERY question in the top 10 Reddit questions is answered.`;

    return { prompt, research };
  }

  generateSectionOutline(questions, intent) {
    const sections = [];
    
    // Group questions by theme
    const themes = {};
    questions.forEach(q => {
      const theme = q.keywords[0] || 'general';
      themes[theme] = themes[theme] || [];
      themes[theme].push(q);
    });

    // Create section for each major theme
    Object.entries(themes).slice(0, 7).forEach(([theme, qs], i) => {
      sections.push(`
   Section ${i + 1}: "${qs[0].title.replace(/\?$/, '')}"
   - Answer the main question comprehensively
   - Address related questions: ${qs.slice(1, 3).map(q => q.title).join('; ')}
   - Include expert perspective
   - Add ${qs.length} practical examples
   - Include data/statistics`);
    });

    return sections.join('\n');
  }

  // ==========================================================================
  // CONTENT GENERATION
  // ==========================================================================

  async generateArticleWithPrompt(prompt) {
    try {
      const result = await this.geminiModel.generateContent({
        contents: [{ role: 'user', parts: [{ text: prompt }] }],
        generationConfig: {
          temperature: 0.7,
          maxOutputTokens: 8192,
          topP: 0.95,
          topK: 40
        }
      });

      const response = await result.response;
      const html = response.text();

      // Extract word count
      const text = html.replace(/<[^>]*>/g, ' ');
      const wordCount = text.split(/\s+/).length;

      return {
        html,
        wordCount,
        generatedAt: new Date().toISOString()
      };

    } catch (error) {
      console.error('Article generation error:', error);
      throw error;
    }
  }

  async addLinks(html, internalPages) {
    // Simplified linking (use full implementation from previous artifact)
    console.log(`   Adding links to ${internalPages.length} internal pages...`);
    
    const prompt = `Add 8-12 strategic internal links to this HTML content.

CONTENT:
${html}

PAGES TO LINK:
${internalPages.map(p => `- ${p.title}: ${p.url} (${p.keywords.join(', ')})`).join('\n')}

Rules: Natural anchor text, spread throughout, minimum 100 chars between links.
Return only the HTML with links added.`;

    try {
      const response = await this.gpt4.chat.completions.create({
        model: 'gpt-4-turbo-preview',
        messages: [
          { role: 'system', content: 'You are an SEO specialist adding internal links.' },
          { role: 'user', content: prompt }
        ],
        temperature: 0.3,
        max_tokens: 8192
      });

      return response.choices[0].message.content;
    } catch (error) {
      console.error('Linking error:', error);
      return html;
    }
  }

  async generateSocialMedia(article, topic, location, research) {
    console.log('   Generating social media posts...');
    
    const topQuestion = research.redditQuestions[0];
    
    const prompt = `Create social media posts for this article about "${topic}".

Article excerpt: ${article.html.substring(0, 500)}...

Key insight from research: The #1 question people ask is "${topQuestion?.title}"

Create:
1. LinkedIn post (1500 chars) - Professional, data-driven
2. Twitter thread (5 tweets) - Engaging, question-based
3. Facebook post (600 chars) - Conversational
4. Instagram caption (2000 chars) - Visual, engaging`;

    // Simplified - use full implementation from previous artifact
    return {
      linkedin: 'LinkedIn post...',
      twitter: 'Twitter thread...',
      facebook: 'Facebook post...',
      instagram: 'Instagram caption...'
    };
  }

  async generatePodcastScript(article, topic, location, research) {
    console.log('   Generating podcast script...');
    return 'Podcast script based on Reddit questions...';
  }

  async generateVideoScript(article, topic, location, research) {
    console.log('   Generating video script...');
    return 'Video script based on Reddit questions...';
  }

  // ==========================================================================
  // QUALITY ASSESSMENT
  // ==========================================================================

  assessResearchQuality(research) {
    const score = {
      experts: Math.min(research.experts.length / 10, 1) * 30,
      questions: Math.min(research.redditQuestions.length / 30, 1) * 40,
      diversity: this.assessDiversity(research) * 30
    };

    const total = Object.values(score).reduce((a, b) => a + b, 0);

    return {
      score: Math.round(total),
      rating: total > 80 ? 'excellent' : total > 60 ? 'good' : total > 40 ? 'fair' : 'poor',
      breakdown: score
    };
  }

  assessDiversity(research) {
    const subreddits = new Set(research.redditQuestions.map(q => q.subreddit));
    const intents = new Set(research.redditQuestions.map(q => q.intent));
    
    return (subreddits.size / 5 + intents.size / 6) / 2;
  }
}

async function main() {
  const factory = new ResearchDrivenContentFactory({
    geminiApiKey: process.env.GEMINI_API_KEY,
    openaiApiKey: process.env.OPENAI_API_KEY,
    braveApiKey: process.env.BRAVE_API_KEY, // Optional for expert search
    brandData: {
      name: 'Digital Growth Agency',
      expertName: 'Sarah Johnson',
      expertTitle: 'Senior SEO Strategist'
    }
  });

  // Generate article with full research
  const result = await factory.generateWithResearch(
    'Best Local SEO Strategies for Restaurants',
    'Austin, TX',
    'local seo restaurants',
    {
      industry: 'Digital Marketing',
      subreddits: ['SEO', 'smallbusiness', 'Entrepreneur', 'restaurantowners'],
      internalPages: [
        {
          url: '/services/local-seo',
          title: 'Local SEO Services',
          keywords: ['local SEO', 'local search', 'Google Business Profile']
        },
        {
          url: '/blog/google-business-profile',
          title: 'Google Business Profile Optimization Guide',
          keywords: ['Google Business Profile', 'GBP', 'local listings']
        }
      ]
    }
  );

  console.log('\n📊 FINAL RESULTS:');
  console.log(`Article: ${result.article.wordCount} words`);
  console.log(`Research Quality: ${result.metadata.researchQuality.rating} (${result.metadata.researchQuality.score}/100)`);
  console.log(`Generation Time: ${result.metadata.generationTime}`);
  
  console.log('\n📚 Research Used:');
  console.log(`- ${result.research.experts.length} experts identified`);
  console.log(`- ${result.research.redditQuestions.length} Reddit questions analyzed`);
  console.log(`- Primary Intent: ${result.research.analysis.primaryIntent}`);
  console.log(`- Target Audience: ${result.research.analysis.targetAudience}`);

  // Save output
  const fs = require('fs').promises;

  await fs.writeFile('research-data.json', JSON.stringify(result.research, null, 2));

  console.log('- research-data.json');
}
