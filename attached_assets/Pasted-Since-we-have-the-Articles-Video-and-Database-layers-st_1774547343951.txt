export async function safeSocialPost(platform: 'linkedin' | 'twitter', content: string, url: string) {
  // 1. CLEANING: Remove any AI-generated artifacts like "Here is your post:"
  const cleanContent = content.replace(/^(Here is|Post:|Social Post:)/i, '').trim();

  // 2. VALIDATION: Ensure the URL is actually a full URL
  if (!url.startsWith('http')) {
    throw new Error("SOCIAL_INVALID_URL: The article URL is missing or malformed.");
  }

  try {
    // 3. THE POST: Use the 'withResilience' wrapper we built earlier
    return await withResilience("Social Post", platform, async () => {
      return await socialClient.post({ text: cleanContent, link: url });
    });
  } catch (error) {
    // Watchdog alerts us if it fails 5 times in a row

  }
}
2. The Podcast "Stutter" Fix (lib/podcast-gen.ts)
Podcasts often "Ghost" because the AI generates a script that is too long for a single OpenAI audio.speech request (which has a 4096-character limit). If the script is 4097 characters, the API returns an error, and the podcast is never created.

The Fix: The "Chunk & Stitch" Method

TypeScript
export async function generateLongPodcast(script: string, jobId: string) {
  const MAX_CHARS = 4000;
  
  // 1. Split script by sentences to stay under the limit
  const chunks = script.match(/[^\.!\?]+[\.!\?]+/g) || [script];
  const audioBuffers: Buffer[] = [];

  try {
    for (const [index, chunk] of chunks.entries()) {
      const response = await openai.audio.speech.create({
        model: "tts-1",
        voice: "alloy",
        input: chunk,
      });
      audioBuffers.push(Buffer.from(await response.arrayBuffer()));
      console.log(`🎙️ Podcast Chunk ${index + 1}/${chunks.length} complete.`);
    }

    // 2. STITCH: Use FFmpeg to merge the chunks into one clean MP3
    const finalBuffer = Buffer.concat(audioBuffers);
    await fs.writeFile(`/tmp/podcast-${jobId}.mp3`, finalBuffer);
    
    return `/tmp/podcast-${jobId}.mp3`;
  } catch (error) {
    await Watchdog.alert("Podcast Agent", `Job ${jobId}`, ["Script too complex or API timeout."]);
    throw error;
  }
}
