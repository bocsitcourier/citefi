The "Command Center" UI Component (components/BulkActionToolbar.tsx)
This is a floating toolbar that appears when you select articles. It turns your "horrible" navigation into a high-speed production line.

TypeScript
import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Checkbox } from '@/components/ui/checkbox';
import { Loader2, ShieldCheck, Zap, Globe } from 'lucide-react';

export function BulkActionToolbar({ selectedIds, onComplete }: { selectedIds: string[], onComplete: () => void }) {
  const [isProcessing, setIsProcessing] = useState(false);
  const [status, setStatus] = useState('');

  const handleBulkHealAndPublish = async () => {
    setIsProcessing(true);
    setStatus('Initializing Platinum Recovery...');

    try {
      // We call a single specialized endpoint that handles the loop on the server
      const response = await fetch('/api/articles/bulk-recover', {
        method: 'POST',
        body: JSON.stringify({ ids: selectedIds }),
      });

      if (response.ok) {
        setStatus('Success! 41 Articles Healed & Published.');
        setTimeout(onComplete, 2000);
      }
    } catch (err) {
      setStatus('Error during bulk recovery.');
    } finally {
      setIsProcessing(false);
    }
  };

  if (selectedIds.length === 0) return null;

  return (
    <div className="fixed bottom-8 left-1/2 -translate-x-1/2 w-[600px] bg-slate-900 text-white p-4 rounded-2xl shadow-2xl border border-purple-500/50 flex items-center justify-between z-50 animate-in fade-in slide-in-from-bottom-4">
      <div className="flex items-center gap-4">
        <div className="bg-purple-600 px-3 py-1 rounded-full text-xs font-bold uppercase tracking-widest">
          {selectedIds.length} Selected
        </div>
        <p className="text-sm text-slate-300 font-medium">
          {isProcessing ? status : 'Ready for Platinum Recovery'}
        </p>
      </div>

      <div className="flex gap-2">
        <Button 
          disabled={isProcessing}
          onClick={handleBulkHealAndPublish}
          className="bg-white text-black hover:bg-purple-50 transition-all font-bold"
        >
          {isProcessing ? <Loader2 className="animate-spin mr-2" /> : <Zap className="mr-2 h-4 w-4 fill-current" />}
          Heal & Publish All
        </Button>
      </div>
    </div>
  );
}
The Backend "Recovery Engine" (app/api/articles/bulk-recover/route.ts)
This is the heavy-duty logic. It pulls the Fuzzy Linker, Surgical Fixer, and Platinum Publisher into one single loop.

TypeScript
export async function POST(req: Request) {
  const { ids } = await req.json();
  
  for (const id of ids) {
    // 1. FUZZY HEAL: Fix those 0-match hyperlinks
    const healedHtml = await applyFuzzyHyperlinks(id);

    // 2. GUARDIAN AUDIT: Check for missing FAQs/Images
    const audit = await auditArticle(healedHtml);
    let finalHtml = healedHtml;

    if (!audit.passed) {
      // 3. SURGICAL FIX: Inject missing pieces without rewriting
      finalHtml = await surgicalFix(healedHtml, audit.missingElements);
    }

    // 4. PLATINUM PUBLISH: Send the Base64 images + HTML to the receiver
    await publishToClient(id, finalHtml);

    // 5. UPDATE DB: Mark as healthy
    await db.update(articles).set({ status: 'published', contentHtml: finalHtml }).where(eq(articles.id, id));
  }

  return Response.json({ success: true });
}
