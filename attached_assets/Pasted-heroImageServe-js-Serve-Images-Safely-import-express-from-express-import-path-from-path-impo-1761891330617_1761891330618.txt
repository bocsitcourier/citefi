heroImageServe.js — Serve Images Safely
import express from "express";
import path from "path";
import fs from "fs";

const router = express.Router();

// Serve local hero images from /public/hero
router.use("/hero", express.static(path.join(process.cwd(), "public", "hero")));

// Optional: Test endpoint for a specific hero image
router.get("/hero/test/:slug", (req, res) => {
  const slug = req.params.slug;
  const filePath = path.join(process.cwd(), "public", "hero", `${slug}.png`);

  if (fs.existsSync(filePath)) {
    res.sendFile(filePath);
  } else {
    console.error(`[HeroImage] Missing local image for slug: ${slug}`);
    res.status(404).send("Hero image not found");
  }
});

export default router;


Usage in your main server:

import heroImageServe from "./heroImageServe.js";

app.use("/", heroImageServe);

2️⃣ heroImageManager.js — Manage Uploads and DB Updates
import fs from "fs";
import path from "path";
import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";
import db from "./db.js"; // Your DB connection

// DigitalOcean Spaces config
const DO_SPACE = "your-space-name";
const DO_REGION = "nyc3";
const DO_ACCESS_KEY = process.env.DO_SPACES_KEY;
const DO_SECRET_KEY = process.env.DO_SPACES_SECRET;
const DO_ENDPOINT = `https://${DO_SPACE}.${DO_REGION}.digitaloceanspaces.com`;

const s3Client = new S3Client({
  region: DO_REGION,
  endpoint: DO_ENDPOINT,
  credentials: {
    accessKeyId: DO_ACCESS_KEY,
    secretAccessKey: DO_SECRET_KEY,
  },
});

// Save image locally AND optionally upload to Spaces
export async function saveHeroImage(slug, buffer, articleId, uploadToDO = true) {
  try {
    const localDir = path.join(process.cwd(), "public/hero");
    if (!fs.existsSync(localDir)) fs.mkdirSync(localDir, { recursive: true });

    const localPath = path.join(localDir, `${slug}.png`);
    fs.writeFileSync(localPath, buffer);
    console.log(`[HeroImage] Saved locally: ${localPath}`);

    let imageUrl = `/hero/${slug}.png`;

    if (uploadToDO) {
      const command = new PutObjectCommand({
        Bucket: DO_SPACE,
        Key: `hero/${slug}.png`,
        Body: buffer,
        ACL: "public-read",
        ContentType: "image/png",
      });

      await s3Client.send(command);
      imageUrl = `${DO_ENDPOINT}/hero/${slug}.png`;
      console.log(`[HeroImage] Uploaded to DO Spaces: ${imageUrl}`);
    }

    // Update DB with hero_image_url
    await db("articles").update({ hero_image_url: imageUrl }).where({ id: articleId });
    console.log(`[HeroImage] DB updated for article ${articleId}`);

    return imageUrl;
  } catch (err) {
    console.error(`[HeroImage] Error saving hero image for slug: ${slug}`, err);
    // Optionally log to admin error log table
    await db("admin_error_logs").insert({
      type: "hero_image_error",
      message: err.message,
      article_id: articleId,
      created_at: new Date(),
    });
    return null;
  }
}

3️⃣ Front-End Integration (Next.js)
import Image from "next/image";

export default function HeroImage({ slug, title }) {
  const url = `/hero/${slug}.png`; // Or full DO URL from DB

  return (
    <Image
      src={url}
      alt={title}
      width={1024}
      height={512}
      onError={(e) => (e.currentTarget.src = "/assets/no-hero.png")}
    />
  );
}

✅ Features Included

Local + DO Spaces support

Automatic folder creation (/public/hero)

Fallback image for missing hero

Database update for hero_image_url

Admin error log insertion for failed saves

Test endpoint to check image availability (/hero/test/:slug)

Console logs for debugging pipeline issues

🧩 Optional Extensions

Concurrent upload for multiple hero images using Promise.all.

Automatic image compression via sharp before saving to reduce file size.

SSR-ready URLs for immediate rendering in the article page.
