The fuzzy-linker.ts doesn't just look for words; it looks for intent. It can "see" through HTML tags and handle variations in how GPT-4 describes your topics (e.g., "Boston MA" vs. "Boston, Massachusetts").

1. The "Fuzzy" Logic
Standard regex fails if your HTML looks like <span>Boston</span> MA. A fuzzy linker treats the HTML as a transparent layer, finds the text match, and then surgically wraps it without breaking the tags.

2. The Platinum Fuzzy Linker (lib/fuzzy-linker.ts)
This script uses a multi-variant regex generator. It takes your Slug Map keywords and automatically creates "fuzzy" variations that GPT-4 is likely to produce.

TypeScript
import cheerio from 'cheerio';

// Helper to create "fuzzy" regex for a phrase
// Example: "Boston MA" -> /Boston,?\s*(MA|Massachusetts)/i
function createFuzzyRegex(keyword: string): RegExp {
  let pattern = keyword
    .replace(/ MA$/i, ",?\\s*(MA|Massachusetts)")
    .replace(/ RI$/i, ",?\\s*(RI|Rhode Island)")
    .replace(/ NH$/i, ",?\\s*(NH|New Hampshire)")
    .replace(/\s+/g, "\\s+"); // Handle extra spaces or newlines

  return new RegExp(`\\b${pattern}\\b`, "i");
}

export function applyFuzzyHyperlinks(html: string, slugMap: { keyword: string, url: string }[]): string {
  const $ = cheerio.load(html, { decodeEntities: false });
  const usedUrls = new Set<string>();

  // Only target text nodes inside content-bearing tags
  $('p, li, td, blockquote').each((_, element) => {
    $(element).contents().each((_, node) => {
      if (node.type === 'text') {
        let text = $(node).text();
        let modified = false;

        for (const entry of slugMap) {
          // 1. Skip if we already linked this URL in this article (Platinum SEO rule)
          if (usedUrls.has(entry.url)) continue;

          const fuzzyRegex = createFuzzyRegex(entry.keyword);
          
          if (fuzzyRegex.test(text)) {
            // 2. Perform the surgical replacement
            const linkedText = text.replace(fuzzyRegex, (match) => {
              usedUrls.add(entry.url);
              modified = true;
              return `<a href="${entry.url}" class="platinum-link">${match}</a>`;
            });

            if (modified) {
              $(node).replaceWith(linkedText);
              return false; // Break the inner loop for this text node
            }
          }
        }
      }
    });
  });

  return $.html();
}
