Refined Node.js FFmpeg Solution (with Audio Fix)
This solution re-uses the two-pass cropdetect method, but specifically adds -c:a copy to the final output options. This tells FFmpeg to copy the original audio stream without re-encoding it, which is the best way to prevent audio sync or cutoff issues.

1. The Code
This code ensures the black bars are cropped, the video is resized to 16:9, and the audio is copied verbatim.

JavaScript

const ffmpeg = require('fluent-ffmpeg');
const path = require('path');

// --- Configuration ---
const INPUT_FILE = 'social-video-15.mp4'; 
const OUTPUT_FILE = path.join(path.dirname(INPUT_FILE), 'final_16-9_no_bars.mp4');

// Target resolution for the final output (16:9 standard)
const TARGET_W = 1920; 
const TARGET_H = 1080;

// --- Main Function to Execute FFmpeg ---

function processVideo() {
    console.log('Starting dynamic crop detection and re-encoding...');
    
    // Pass 1: Run cropdetect to find the content area
    const proc = ffmpeg(INPUT_FILE)
        .noVideo()
        .outputOptions('-vf', 'cropdetect=limit=0.01:round=16')
        
    let cropFilter = null;

    proc.on('stderr', (stderrLine) => {
        // Look for the 'crop=W:H:X:Y' filter in the stderr output
        const cropMatch = stderrLine.match(/crop=(\d+:\d+:\d+:\d+)/);
        if (cropMatch) {
            cropFilter = cropMatch[0];
        }
    });

    proc.on('error', (err) => {
        // This process might be killed before finishing, which throws an error.
        // We only care about errors not related to premature kill.
        if (!err.message.includes('SIGKILL') && !err.message.includes('SIGINT')) {
            console.error(`❌ Crop Detection Error: ${err.message}`);
        }
    });

    proc.on('end', () => {
        if (!cropFilter) {
            console.error('❌ Could not detect crop area. Are the black bars pure black?');
            // Fallback: If cropdetect fails, assume no internal padding and use the simplest crop/scale
            cropFilter = 'null'; 
        }

        // Pass 2: Apply the crop and scale/re-encode
        reEncodeWithCrop(cropFilter);
    });

    // Execute Pass 1
    proc.run();
}

/**
 * Re-encodes the video using the detected crop filter and scales it to 16:9.
 * @param {string} filter - The 'crop=W:H:X:Y' string or 'null' for fallback.
 */
function reEncodeWithCrop(filter) {
    console.log(`\nStarting Re-encode with Filter: ${filter === 'null' ? 'Simple Crop/Scale (Fallback)' : filter}`);
    
    let videoFilters = [];

    // 1. Apply the detected crop filter to remove the baked-in black bars
    if (filter !== 'null') {
        videoFilters.push(filter);
    }

    // 2. Scale the remaining content to FILL the 16:9 target size
    // We use increase to ensure the shortest side of the content fills the shortest side of the container.
    videoFilters.push(`scale=${TARGET_W}:-1:force_original_aspect_ratio=increase`);
    
    // 3. Crop the scaled content to the exact 16:9 size, centered
    videoFilters.push(`crop=${TARGET_W}:${TARGET_H}`);


    ffmpeg(INPUT_FILE)
        .videoFilter(videoFilters)
        .outputOptions([
            // Video codec and quality settings
            '-c:v libx264',
            '-crf 23', 
            
            // 🔥 AUDIO FIX: Copy the original audio stream directly to prevent cutoffs
            '-c:a copy' 
        ])
        .on('end', () => {
            console.log(`\n✅ **SUCCESS!** Video processed and saved to ${OUTPUT_FILE}`);
        })
        .on('error', (err) => {
            console.error(`\n❌ **FATAL RE-ENCODE ERROR:** ${err.message}`);
        })
        .save(OUTPUT_FILE);
}

processVideo();
Key Changes & Debugging Notes
Audio Fix (-c:a copy): By using '-c:a copy' in the output options, we instruct FFmpeg to simply copy the audio stream from the input to the output without re-encoding it. This nearly eliminates audio cutoff issues caused by codec or stream duration mismatches during the video processing.

Robust cropdetect: The code now handles the possibility that cropdetect might fail to find the filter (which happens if the black "bars" aren't perfectly black) and falls back to the simple Crop/Scale method you tried before.

Correct Filter Order: The sequence is crucial: a. Remove Baked-in Bars (crop=W:H:X:Y) b. Scale Content (scale=...:increase) c. Final Crop (crop=1920:1080)

If this dynamic cropdetect method still fails to remove the black bars, it means the issue is complex, and you may need to manually inspect the content dimensions using a tool like Mediainfo to determine the exact padding values used by the Gemini API output, and then use a fixed crop filter instead of the dynamic one.
