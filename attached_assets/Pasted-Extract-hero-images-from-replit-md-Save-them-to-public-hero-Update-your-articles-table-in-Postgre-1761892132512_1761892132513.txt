Extract hero images from replit.md

Save them to /public/hero

Update your articles table in PostgreSQL

Make them immediately displayable in SSR pages

1️⃣ Setup Dependencies
npm install fs path pg


fs and path are built-in.

pg is for PostgreSQL updates.

2️⃣ Create heroImagePipeline.js
import fs from "fs";
import path from "path";
import { Client } from "pg";

// --- Config ---
const REPLIT_MD_PATH = path.join(process.cwd(), "replit.md");
const HERO_DIR = path.join(process.cwd(), "public/hero");

// PostgreSQL config
const dbClient = new Client({
  user: process.env.DB_USER,
  host: process.env.DB_HOST,
  database: process.env.DB_NAME,
  password: process.env.DB_PASSWORD,
  port: process.env.DB_PORT,
});

// Ensure hero directory exists
if (!fs.existsSync(HERO_DIR)) fs.mkdirSync(HERO_DIR, { recursive: true });

// --- Function to extract base64 images from replit.md ---
function extractImages(mdContent) {
  // Example: images in format: ![hero](data:image/png;base64,...)
  const regex = /!\[.*?\]\((data:image\/.*?;base64,.*?)\)/g;
  const matches = [];
  let match;
  while ((match = regex.exec(mdContent)) !== null) {
    matches.push(match[1]); // push base64 string
  }
  return matches;
}

// --- Save image to /public/hero ---
function saveImage(base64Data, slug) {
  const cleanBase64 = base64Data.replace(/^data:image\/\w+;base64,/, "");
  const buffer = Buffer.from(cleanBase64, "base64");
  const filePath = path.join(HERO_DIR, `${slug}.png`);
  fs.writeFileSync(filePath, buffer);
  return `/hero/${slug}.png`; // path to store in DB
}

// --- Update PostgreSQL article ---
async function updateArticleHero(articleId, heroUrl) {
  const query = `UPDATE articles SET hero_image_url=$1 WHERE id=$2`;
  await dbClient.query(query, [heroUrl, articleId]);
}

// --- Main Pipeline ---
async function runPipeline() {
  await dbClient.connect();

  const mdContent = fs.readFileSync(REPLIT_MD_PATH, "utf-8");
  const images = extractImages(mdContent);

  console.log(`Found ${images.length} hero images.`);

  for (let i = 0; i < images.length; i++) {
    const slug = `article-${i + 1}`; // Replace with your real slug logic
    const heroUrl = saveImage(images[i], slug);

    // Replace with your real article IDs
    const articleId = i + 1;
    await updateArticleHero(articleId, heroUrl);

    console.log(`Saved and linked hero image for article ${articleId}`);
  }

  await dbClient.end();
  console.log("Hero image pipeline complete!");
}

// Run if executed directly
if (require.main === module) {
  runPipeline().catch(console.error);
}

3️⃣ Display in SSR Page
import Image from "next/image";

export default function ArticleHero({ heroImageUrl, title }) {
  if (!heroImageUrl) return null;
  return (
    <Image
      src={heroImageUrl}
      alt={title}
      width={1024}
      height={512}
      priority
      onError={(e) => (e.currentTarget.src = "/assets/no-hero.png")}
    />
  );
}

4️⃣ Usage

Place replit.md in your project root.

Make sure it contains image references in base64 format:

![hero](data:image/png;base64,iVBORw0KGgoAAAANS...)


Run the pipeline:

node heroImagePipeline.js


Check /public/hero — images should now be there.

Open your Next.js page — the hero images should display.
