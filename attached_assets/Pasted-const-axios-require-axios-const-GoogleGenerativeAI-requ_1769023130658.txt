const axios = require('axios');
const { GoogleGenerativeAI } = require('@google/generative-ai');

class SmartTopicGenerator {
  constructor(config = {}) {
    this.config = {
      maxTopics: config.maxTopics || 20,
      modelName: 'gemini-2.0-flash-exp', 
      braveApiKey: config.braveApiKey || process.env.BRAVE_SEARCH_API_KEY,
      geminiApiKey: config.geminiApiKey || process.env.GEMINI_API_KEY,
    };
    
    this.genAI = new GoogleGenerativeAI(this.config.geminiApiKey);
    // Specialized models for different tasks
    this.analyst = this.genAI.getGenerativeModel({ 
        model: this.config.modelName, 
        generationConfig: { temperature: 0.1 } 
    });
    this.creative = this.genAI.getGenerativeModel({ 
        model: this.config.modelName, 
        generationConfig: { temperature: 0.8 } 
    });
  }

  /**
   * 1. PARALLEL PIPELINE
   */
  async generateTopics(seedTitle, options = {}) {
    this.log('🚀 Initializing Parallel Intelligence Pipeline...');

    try {
      // Execute data gathering tasks in parallel
      const [intentData, searchResults, entities] = await Promise.all([
        this.analyzeIntent(seedTitle, options),
        this.getSearchData(seedTitle, options),
        options.location ? this.findNearbyEntities(options.location, options.industry) : Promise.resolve([])
      ]);

      // 2. CRITIQUE LOOP (The "Smart" Step)
      const draftTopics = await this.generateDrafts(seedTitle, { intentData, searchResults, entities }, options);
      const refinedTopics = await this.critiqueAndRefine(draftTopics, searchResults, options);

      // 3. MATHEMATICAL SCORING
      const finalTopics = refinedTopics.map(topic => ({
        ...topic,
        uniquenessScore: this.calculateJaccardUniqueness(topic.title, searchResults.map(r => r.title))
      }));

      return this.formatFinalReport(finalTopics, seedTitle, options);
    } catch (error) {
      this.log('❌ Pipeline Error', { error: error.message });
      throw error;
    }
  }

  /**
   * 2. AGENTIC CRITIQUE (Self-Correction)
   */
  async critiqueAndRefine(drafts, competitors, options) {
    this.log('🧠 Agentic Review: Checking for hallucinations and AI-patterns...');
    
    const prompt = `
      You are a Senior Content Strategist. Review these drafted titles for a ${options.industry} client.
      
      DRAFTS: ${JSON.stringify(drafts)}
      COMPETITORS: ${JSON.stringify(competitors.slice(0, 5).map(c => c.title))}

      CRITIQUE CRITERIA:
      1. Remove "Hallucinations": Flag services ${options.industry} doesn't actually provide.
      2. Kill Clichés: Remove titles starting with "The Ultimate Guide to..." or "Unlocking the Secrets...".
      3. GEO-Alignment: Ensure city names and local entities feel natural, not forced.

      Return ONLY the refined JSON list of topics.
    `;

    const result = await this.analyst.generateContent(prompt);
    return JSON.parse(result.response.text());
  }

  /**
   * 3. JACCARD SIMILARITY (Vector-like Uniqueness)
   * Ensures the code doesn't suggest topics already on Google Page 1.
   */
  calculateJaccardUniqueness(candidateTitle, competitorTitles) {
    const getSet = (str) => new Set(str.toLowerCase().replace(/[^\w\s]/g, '').split(/\s+/));
    const candidateSet = getSet(candidateTitle);
    
    let highestSimilarity = 0;

    competitorTitles.forEach(comp => {
      const compSet = getSet(comp);
      const intersection = new Set([...candidateSet].filter(x => compSet.has(x)));
      const union = new Set([...candidateSet, ...compSet]);
      const similarity = intersection.size / union.size;
      if (similarity > highestSimilarity) highestSimilarity = similarity;
    });

    // Score: 100 is perfectly unique, 0 is a direct copy
    return Math.round((1 - highestSimilarity) * 100);
  }

  log(msg) { console.log(`[${new Date().toISOString()}] ${msg}`); }
}
