The Retrospective Engine (lib/retrospective.ts)This script looks at the "Guardian Audit" reports for your failed articles and synthesizes a "Lesson Plan."TypeScriptimport { db } from "../lib/db";
import { OpenAI } from "openai";

export async function runRetrospective(failedArticleIds: string[]) {
  // 1. Fetch the Guardian's audit logs for these failures
  const audits = await db.article_audits.findMany({
    where: { articleId: { in: failedArticleIds } }
  });

  const failurePatterns = audits.map(a => a.issues.join(", ")).join("\n");

  // 2. Ask GPT-4 to find the "Root Cause" in the instructions
  const openai = new OpenAI();
  const analysis = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      { 
        role: "system", 
        content: "You are a Meta-Prompt Engineer. Analyze these content failures and identify exactly which part of the MASTER_PROMPT is causing the AI to skip FAQs or images." 
      },
      { 
        role: "user", 
        content: `Here are the issues found in 41 failed articles:\n${failurePatterns}` 
      }
    ],
    response_format: { type: "json_object" }
  });

  const lesson = JSON.parse(analysis.choices[0].message.content || "{}");

  // 3. Save to the "AI Learning Center" (Style Ledger)
  await db.ai_learning_ledger.create({
    data: {
      category: "SYSTEM_RECALL",
      insight: lesson.root_cause,
      correction: lesson.suggested_instruction_tweak,
      priority: "HIGH"
    }
  });

  return lesson;
}
