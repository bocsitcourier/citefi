Complete Veo 3 + Voice Synthesis Workflow
Overview
The workflow involves:

Script & Voice Planning - Write dialogue with voice direction
Veo 3 Video Generation - Create video with native lip-sync
Voice Replacement (Optional) - Replace with ElevenLabs or custom voice for consistency
Audio Humanization - Add natural speech patterns
Final Assembly - Sync and polish

interface SceneSegment {
  id: number;
  startTime: number;
  duration: number;
  visualPrompt: string;
  dialogue: string;
  voiceDirection: string;
  negativePrompt: string;
  characterConsistency: string;
}

interface VeoPrompt {
  mainPrompt: string;
  negativePrompt: string;
  duration: number;
  aspectRatio: '16:9' | '9:16' | '1:1';
  style?: string;
}

class VeoPromptGenerator {
  private characterDescription: string;
  private voiceProfile: string;
  private styleGuide: string;

  constructor(config: {
    character: string;
    voice: string;
    style: string;
  }) {
    this.characterDescription = config.character;
    this.voiceProfile = config.voice;
    this.styleGuide = config.style;
  }

  generateInitialPrompt(scene: SceneSegment): VeoPrompt {
    return {
      mainPrompt: `
${this.styleGuide}

${this.characterDescription}

Scene: ${scene.visualPrompt}

The character says: "${scene.dialogue}"
Voice delivery: ${scene.voiceDirection}

Camera: ${this.inferCameraWork(scene)}
Lighting: Cinematic, natural
      `.trim(),
      
      negativePrompt: `
(no subtitles, no captions, no text overlay, no watermark)
(no background music, no sound effects unless specified)
(no blurry faces, no distorted speech, no robotic voice)
(no jump cuts, no glitches)
      `.trim(),
      
      duration: scene.duration,
      aspectRatio: '16:9'
    };
  }

  generateExtensionPrompt(
    previousScene: SceneSegment, 
    currentScene: SceneSegment
  ): VeoPrompt {
    return {
      mainPrompt: `
Continue seamlessly from the previous scene.

${this.characterDescription}

Next action: ${currentScene.visualPrompt}

The character continues: "${currentScene.dialogue}"
Voice delivery: ${currentScene.voiceDirection}

Maintain exact same:
- Character appearance
- Voice tone and timbre
- Lighting conditions
- Camera style
      `.trim(),
      
      negativePrompt: `
(no subtitles, no captions, no text overlay)
(no background music)
(no voice change, no accent shift)
(no costume change, no lighting change)
(no scene break, continuous action)
      `.trim(),
      
      duration: currentScene.duration,
      aspectRatio: '16:9'
    };
  }

  private inferCameraWork(scene: SceneSegment): string {
    if (scene.visualPrompt.toLowerCase().includes('close-up')) {
      return 'Close-up shot, shallow depth of field';
    }
    if (scene.visualPrompt.toLowerCase().includes('wide')) {
      return 'Wide establishing shot';
    }
    return 'Medium shot, eye level, slight movement';
  }
}

class VeoChainedExtension {

  private totalDuration: number = 0;
  private targetDuration: number = 60;
  private maxDuration: number = 148; // Veo 3.1 limit

  async generateFullVideo(scenes: SceneSegment[]): Promise<string> {

    for (let i = 0; i < scenes.length; i++) {
      const scene = scenes[i];
      
      if (this.totalDuration >= this.targetDuration) {
        console.log('✅ Target duration reached');
        break;
      }

      if (i === 0) {
        // Generate initial clip
        await this.generateInitialClip(scene);
      } else {
        // Extend from previous clip
        await this.extendFromPrevious(scene);
      }

      console.log(`  Clip ${i + 1}: ${this.totalDuration}s total`);
    }

    // Combine all clips
    return await this.combineClips();
  }

  private async generateInitialClip(scene: SceneSegment): Promise<void> {
    const prompt = this.buildPrompt(scene, null);
    
    // This would call the actual Veo API
    const result = await this.callVeoAPI({

4. Voice Humanization & Replacement System
If Veo's native voice isn't consistent enough, use this voice replacement workflow:
typescript// voice-humanization.ts

interface VoiceProfile {
  id: string;
  name: string;
  baseVoice: string; // ElevenLabs voice ID or custom model
  characteristics: {
    pitch: number;        // -20 to +20 semitones
    speed: number;        // 0.5 to 2.0
    stability: number;    // 0 to 1 (lower = more expressive)
    similarity: number;   // 0 to 1 (voice match strength)
    style: number;        // 0 to 1 (exaggeration)
  };
  humanizationRules: HumanizationRule[];
}

interface HumanizationRule {
  type: 'pause' | 'breath' | 'filler' | 'emphasis' | 'variation';
  trigger: string | RegExp;
  effect: any;
}

class VoiceHumanizer {
  private profile: VoiceProfile;

  constructor(profile: VoiceProfile) {
    this.profile = profile;
  }

  // Add natural speech patterns to text
  humanizeScript(text: string): string {
    let humanized = text;

    // Add micro-pauses after punctuation
    humanized = humanized.replace(/\.\s/g, '. <break time="300ms"/> ');
    humanized = humanized.replace(/,\s/g, ', <break time="150ms"/> ');
    
    // Add breath marks for long sentences
    humanized = this.addBreathMarks(humanized);
    
    // Add emphasis markers
    humanized = this.addEmphasis(humanized);
    
    // Add natural fillers sparingly
    humanized = this.addFillers(humanized);
    
    // Add pitch variations
    humanized = this.addPitchVariation(humanized);

    return humanized;
  }

  private addBreathMarks(text: string): string {
    const sentences = text.split(/(?<=[.!?])\s+/);
    
    return sentences.map(sentence => {
      const words = sentence.split(' ');
      
      // Add breath every 8-12 words
      if (words.length > 10) {
        const midpoint = Math.floor(words.length / 2);
        words.splice(midpoint, 0, '<break time="200ms" strength="weak"/>');
      }
      
      return words.join(' ');
    }).join(' ');
  }

  private addEmphasis(text: string): string {
    // Emphasize key words (adjectives, important nouns)
    const emphasisWords = ['secret', 'perfect', 'important', 'always', 'never', 'must'];
    
    let result = text;
    emphasisWords.forEach(word => {
      const regex = new RegExp(`\\b(${word})\\b`, 'gi');
      result = result.replace(regex, '<emphasis level="moderate">$1</emphasis>');
    });
    
    return result;
  }

  private addFillers(text: string): string {
    // Add occasional natural fillers (very sparingly)
    const fillers = ['um', 'you know', 'well'];
    
    // Only add fillers to conversational content, not narration
    if (!text.includes('"')) return text;
    
    // Add a filler roughly once per 50 words
    const words = text.split(' ');
    if (words.length > 50 && Math.random() > 0.7) {
      const insertPoint = Math.floor(words.length * 0.3);
      const filler = fillers[Math.floor(Math.random() * fillers.length)];
      words.splice(insertPoint, 0, `${filler},`);
    }
    
    return words.join(' ');
  }

  private addPitchVariation(text: string): string {
    // Questions should rise in pitch
    let result = text.replace(
      /([^.!?]*\?)/g, 
      '<prosody pitch="+5%">$1</prosody>'
    );
    
    // Exclamations should have more energy
    result = result.replace(
      /([^.!?]*!)/g,
      '<prosody rate="105%" pitch="+3%">$1</prosody>'
    );
    
    return result;
  }

  // Generate SSML for voice synthesis
  toSSML(text: string): string {
    const humanized = this.humanizeScript(text);
    
    return `
<speak version="1.0" xmlns="http://www.w3.org/2001/10/synthesis">
  <voice name="${this.profile.baseVoice}">
    <prosody rate="${this.profile.characteristics.speed * 100}%" pitch="${this.profile.characteristics.pitch > 0 ? '+' : ''}${this.profile.characteristics.pitch}%">
      ${humanized}
    </prosody>
  </voice>
</speak>
    `.trim();
  }
}

// ElevenLabs Integration
class ElevenLabsVoiceGenerator {
  private apiKey: string;
  private voiceId: string;

  constructor(apiKey: string, voiceId: string) {
    this.apiKey = apiKey;
    this.voiceId = voiceId;
  }

  async generateSpeech(
    text: string, 
    settings: {
      stability: number;
      similarityBoost: number;
      style: number;
      useSpeakerBoost: boolean;
    }
  ): Promise<Buffer> {
    const response = await fetch(
      `https://api.elevenlabs.io/v1/text-to-speech/${this.voiceId}`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'xi-api-key': this.apiKey
        },
        body: JSON.stringify({
          text,
          model_id: 'eleven_multilingual_v2',
          voice_settings: {
            stability: settings.stability,
            similarity_boost: settings.similarityBoost,
            style: settings.style,
            use_speaker_boost: settings.useSpeakerBoost
          }
        })
      }
    );

    return Buffer.from(await response.arrayBuffer());
  }

  // Generate with humanization
  async generateHumanizedSpeech(
    text: string,
    humanizer: VoiceHumanizer
  ): Promise<Buffer> {
    const humanizedText = humanizer.humanizeScript(text);
    
    return this.generateSpeech(humanizedText, {
      stability: 0.3,        // Lower = more expressive
      similarityBoost: 0.8,  // High voice consistency
      style: 0.4,            // Moderate style exaggeration
      useSpeakerBoost: true
    });
  }
}

5. Complete Video Assembly Pipeline
typescript// video-assembly-pipeline.ts

import { execSync } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';

interface VideoSegment {
  videoPath: string;
  originalAudioPath: string;
  replacementAudioPath?: string;
  startTime: number;
  duration: number;
}

class VideoAssemblyPipeline {
  private workDir: string;
  private segments: VideoSegment[] = [];

  constructor(workDir: string = './temp_assembly') {
    this.workDir = workDir;
    if (!fs.existsSync(workDir)) {
      fs.mkdirSync(workDir, { recursive: true });
    }
  }

  // Step 1: Extract audio timing from Veo video
  extractAudioTiming(videoPath: string): { 
    speechSegments: Array<{ start: number; end: number; text?: string }> 
  } {
    // Use Whisper or similar for speech detection
    const outputPath = path.join(this.workDir, 'timing.json');
    
    execSync(`
      whisper "${videoPath}" \
        --model medium \
        --output_format json \
        --output_dir "${this.workDir}"
    `);

    const transcription = JSON.parse(fs.readFileSync(outputPath, 'utf-8'));
    
    return {
      speechSegments: transcription.segments.map((seg: any) => ({
        start: seg.start,
        end: seg.end,
        text: seg.text
      }))
    };
  }

  // Step 2: Generate replacement audio with matching timing
  async generateMatchedAudio(
    segments: Array<{ start: number; end: number; text: string }>,
    voiceGenerator: ElevenLabsVoiceGenerator,
    humanizer: VoiceHumanizer
  ): Promise<string[]> {
    const audioPaths: string[] = [];

    for (let i = 0; i < segments.length; i++) {
      const segment = segments[i];
      const targetDuration = segment.end - segment.start;
      
      // Generate speech
      const audioBuffer = await voiceGenerator.generateHumanizedSpeech(
        segment.text,
        humanizer
      );

      const audioPath = path.join(this.workDir, `segment_${i}.mp3`);
      fs.writeFileSync(audioPath, audioBuffer);

      // Adjust timing to match original
      const adjustedPath = await this.adjustAudioTiming(
        audioPath, 
        targetDuration
      );
      
      audioPaths.push(adjustedPath);
    }

    return audioPaths;
  }

  // Step 3: Adjust audio speed to match lip-sync timing
  private async adjustAudioTiming(
    audioPath: string, 
    targetDuration: number
  ): Promise<string> {
    const outputPath = audioPath.replace('.mp3', '_adjusted.mp3');
    
    // Get current duration
    const probeResult = execSync(
      `ffprobe -v error -show_entries format=duration -of csv=p=0 "${audioPath}"`
    ).toString().trim();
    
    const currentDuration = parseFloat(probeResult);
    const speedFactor = currentDuration / targetDuration;

    // Use rubberband for high-quality time-stretching (preserves pitch)
    if (Math.abs(speedFactor - 1) > 0.05) {
      execSync(`
        ffmpeg -i "${audioPath}" \
          -filter:a "rubberband=tempo=${speedFactor}" \
          -y "${outputPath}"
      `);
      return outputPath;
    }

    return audioPath;
  }

  // Step 4: Replace audio in video while preserving lip-sync
  async replaceAudioTrack(
    videoPath: string,
    newAudioSegments: Array<{ path: string; startTime: number }>,
    outputPath: string
  ): Promise<void> {
    // Create a complex filter for audio replacement
    const filterComplex = this.buildAudioReplacementFilter(newAudioSegments);

    // Build FFmpeg command
    let inputArgs = `-i "${videoPath}"`;
    newAudioSegments.forEach((seg, i) => {
      inputArgs += ` -i "${seg.path}"`;
    });

    execSync(`
      ffmpeg ${inputArgs} \
        -filter_complex "${filterComplex}" \
        -map 0:v -map "[aout]" \
        -c:v copy \
        -c:a aac -b:a 192k \
        -y "${outputPath}"
    `);
  }

  private buildAudioReplacementFilter(
    segments: Array<{ path: string; startTime: number }>
  ): string {
    // Build FFmpeg filter to mix audio segments at correct times
    const delays = segments.map((seg, i) => 
      `[${i + 1}:a]adelay=${Math.round(seg.startTime * 1000)}|${Math.round(seg.startTime * 1000)}[a${i}]`
    ).join(';');

    const mixInputs = segments.map((_, i) => `[a${i}]`).join('');
    const mix = `${mixInputs}amix=inputs=${segments.length}:duration=longest[aout]`;

    return `${delays};${mix}`;
  }

  // Step 5: Add ambient audio / background music
  async addAmbientAudio(
    videoPath: string,
    ambientPath: string,
    outputPath: string,
    ambientVolume: number = 0.15
  ): Promise<void> {
    execSync(`
      ffmpeg -i "${videoPath}" -i "${ambientPath}" \
        -filter_complex "[0:a][1:a]amix=inputs=2:duration=first:weights=1 ${ambientVolume}[aout]" \
        -map 0:v -map "[aout]" \
        -c:v copy -c:a aac \
        -y "${outputPath}"
    `);
  }

  // Complete pipeline
  async runFullPipeline(config: {
    veoVideoPath: string;
    voiceGenerator: ElevenLabsVoiceGenerator;
    humanizer: VoiceHumanizer;
    ambientAudioPath?: string;
    outputPath: string;
  }): Promise<string> {
    console.log('🎬 Starting video assembly pipeline...\n');

    // Step 1: Extract timing
    console.log('1️⃣ Extracting speech timing from Veo video...');
    const timing = this.extractAudioTiming(config.veoVideoPath);
    console.log(`   Found ${timing.speechSegments.length} speech segments\n`);

    // Step 2: Generate replacement audio
    console.log('2️⃣ Generating humanized voice audio...');
    const audioPaths = await this.generateMatchedAudio(
      timing.speechSegments.filter(s => s.text),
      config.voiceGenerator,
      config.humanizer
    );
    console.log(`   Generated ${audioPaths.length} audio segments\n`);

    // Step 3: Replace audio track
    console.log('3️⃣ Replacing audio track in video...');
    const intermediateOutput = path.join(this.workDir, 'with_voice.mp4');
    await this.replaceAudioTrack(
      config.veoVideoPath,
      audioPaths.map((p, i) => ({ 
        path: p, 
        startTime: timing.speechSegments[i].start 
      })),
      intermediateOutput
    );

    // Step 4: Add ambient audio if provided
    if (config.ambientAudioPath) {
      console.log('4️⃣ Adding ambient audio...');
      await this.addAmbientAudio(
        intermediateOutput,
        config.ambientAudioPath,
        config.outputPath
      );
    } else {
      fs.copyFileSync(intermediateOutput, config.outputPath);
    }

    console.log(`\n✅ Complete! Output: ${config.outputPath}`);
    return config.outputPath;
  }
}
