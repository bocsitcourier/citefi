To implement the "Image Memory" system, your worker needs a new pre-generation step. Instead of going straight to the AI, it will run a "Discovery Query" to see if a suitable image already exists.

🛠️ The "Image Discovery" Utility (lib/image-memory.ts)
This function searches your database for the most relevant existing image based on the article's keywords.

TypeScript
import { db } from "./db";
import { articles } from "./schema";
import { eq, and, isNotNull, ilike, or } from "drizzle-orm";

/**
 * 🧠 Image Memory Search
 * Looks for an existing high-quality image from a previous article
 * that matches the current topic.
 */
export async function findExistingImage(keywords: string[], title: string) {
  // 1. Get the primary keyword (usually the first one)
  const primaryKeyword = keywords[0];

  if (!primaryKeyword) return null;

  // 2. Search for a 'COMPLETED' article with a similar keyword or title
  const match = await db.query.articles.findFirst({
    where: and(
      eq(articles.status, "COMPLETED"),
      isNotNull(articles.hero_image_url),
      or(
        ilike(articles.keywords, `%${primaryKeyword}%`),
        ilike(articles.title, `%${title.substring(0, 15)}%`)
      )
    ),
    orderBy: (articles, { desc }) => [desc(articles.updatedAt)],
  });

  return match?.hero_image_url || null;
}

// Inside your Image Generation Worker
export async function handleImageGeneration(articleId: string, keywords: string[], title: string) {
  
  // STEP 1: Check Memory (Cost: $0.00)
  const existingUrl = await findExistingImage(keywords, title);
  
  if (existingUrl) {
    console.log(`♻️ [Job ${articleId}] Reusing existing image: ${existingUrl}`);
    await db.update(articles)
      .set({ hero_image_url: existingUrl })
      .where(eq(articles.id, articleId));
