This script is your "System Heartbeat" for the 2.5 migration. It bypasses the complex worker logic and talks directly to the new engine to ensure the brand integrity and style rules are being followed before you flip the switch on 1,000 articles.🧪 The 2.5 Migration Verifier (scripts/verify-2.5-migration.ts)This script tests three things: Latency (to ensure 2.5 is faster), Brand Integrity (to catch hallucinations), and Visual Prompting (to ensure no infographics).TypeScriptimport { GoogleGenerativeAI } from "@google/generative-ai";
import { enforceBrandIntegrity } from "../lib/text-sanitizer"; // Our brand guard
import { generateVisualPrompt } from "../lib/prompt-engine"; // Our "Happy People" guard

const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY!);

async function verifyMigration() {
  console.log("🚀 Starting Gemini 2.5 Flash Migration Test...");
  
  // 1. Model Selection (The new Enterprise Stable)
  const model = genAI.getGenerativeModel({ model: "gemini-2.5-flash" });

  const testTopic = "Assisted living benefits for seniors in Weston";
  const startTime = performance.now();

  try {
    // 2. The Test Prompt
    const prompt = `Write a short 50-word introduction about ${testTopic}. 
    Ensure you mention the company name: Senior Care Weston.`;

    const result = await model.generateContent(prompt);
    const response = await result.response;
    const rawText = response.text();
    const endTime = performance.now();

    // 3. ANALYSIS: Latency
    const duration = ((endTime - startTime) / 1000).toFixed(2);
    console.log(`⏱️ Latency: ${duration}s (Expected < 5s for 2.5 Flash)`);

    // 4. ANALYSIS: Brand Integrity
    const sanitizedText = enforceBrandIntegrity(rawText);
    const hallucinationDetected = rawText.toLowerCase().includes("caregiver weston") && !rawText.toLowerCase().includes("senior care weston");
    
    console.log("📝 Raw Output Preview:", rawText.substring(0, 100) + "...");
    
    if (hallucinationDetected) {
      console.error("❌ BRAND ALERT: AI tried to use a generic name. Sanitizer caught it.");
    } else {
      console.log("✅ BRAND INTEGRITY: Passed.");
    }

    // 5. ANALYSIS: Visual Prompting
    const visualPrompt = generateVisualPrompt("A happy senior playing chess");
    console.log("🎨 Visual Prompt Check:", visualPrompt.includes("vibrant") ? "✅ Vibrant Style Included" : "❌ Style Missing");
    
    if (visualPrompt.includes("infographic")) {
      console.error("❌ VISUAL ALERT: Infographics still allowed in prompt!");
    } else {
      console.log("✅ VISUAL INTEGRITY: Infographics blocked.");
    }

    console.log("\n🏁 Conclusion: Gemini 2.5 Flash is ready for Production.");

  } catch (error) {
    console.error("💥 Migration Test Failed:", error.message);
  }
}

verifyMigration();
