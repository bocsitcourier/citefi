import { statelessDb } from "./db";
import { articles } from "./schema";
import { eq } from "drizzle-orm";
// Assume imports for your context fetchers, gemini generators, and parallelEnrichmentStage

// ─── UTILITIES ─────────────────────────────────────────────────────────────

/**
 * 🚦 JITTER: Random delay to prevent the "Thundering Herd" API crash.
 */
async function applyJitter(minMs = 5000, maxMs = 25000) {
  const delay = Math.floor(Math.random() * (maxMs - minMs + 1) + minMs);
  console.log(`🚦 Staggering API call: waiting ${(delay / 1000).toFixed(1)}s...`);
  return new Promise((resolve) => setTimeout(resolve, delay));
}

/**
 * 🕵️‍♂️ BACKGROUND CRITIQUE: Fire-and-forget logic to prevent blocking.
 */
function launchBackgroundCritique(articleId: string, content: string) {
  console.log(`🕵️‍♂️ Launching background critique for ${articleId}... (Not blocking)`);
  
  // Notice there is NO 'await' here. This runs silently in the background.
  runArticleCritique(content)
    .then(async (score) => {
      await statelessDb.update(articles)
        .set({ quality_score: score })
        .where(eq(articles.id, articleId));
      console.log(`📊 Background score saved for ${articleId}: ${score}/100`);
    })
    .catch((err) => {
      console.error(`⚠️ Background critique failed for ${articleId}, ignoring.`, err.message);
    });
}

// ─── MAIN PIPELINE ─────────────────────────────────────────────────────────

export async function processArticleJob(articleId: string) {
  try {
    console.log(`🚀 [Job ${articleId}] Starting Enterprise Pipeline...`);

    // 1. UPDATE DB: Status -> PROCESSING (Using Stateless DB)
    await statelessDb.update(articles)
      .set({ status: 'PROCESSING', progress_percent: 10, current_step_label: "🔍 Gathering Context" })
      .where(eq(articles.id, articleId));

    // 2. PARALLEL CONTEXT FETCH: Get SEO and Guardian warnings simultaneously
    const [guardianWarnings, seoContext] = await Promise.all([
      getGuardianFailureWarnings(articleId),
      fetchSEOContext(articleId)
    ]);

    // 3. APPLY JITTER: Spread out the 10+ concurrent API requests
    await applyJitter(5000, 25000);

    // 4. CORE GENERATION: Call Gemini 2.5 Flash
    await statelessDb.update(articles)
      .set({ progress_percent: 30, current_step_label: "✍️ Generating Core Content (Gemini 2.5 Flash)" })
      .where(eq(articles.id, articleId));

    const rawDraft = await generateCoreWithGemini(articleId, guardianWarnings, seoContext);

    // 5. FIRE-AND-FORGET CRITIQUE: Instantly launch Stage 1.5 and move on
    launchBackgroundCritique(articleId, rawDraft.body);

    // 6. PARALLEL ENRICHMENT: Fan-out Images, HTML Polish, and SEO Metadata
    await statelessDb.update(articles)
      .set({ progress_percent: 70, current_step_label: "🎨 Parallel Enrichment (Images + HTML + SEO)" })
      .where(eq(articles.id, articleId));

    const { images, finalizedHtml, seoData } = await parallelEnrichmentStage(
      articleId,
      rawDraft.body,
      {
        imagePrompts: rawDraft.imagePrompts,
        brandName: "Senior Care Weston", // Hardcoded brand integrity
        keywords: seoContext.keywords
      }
    );

    // 7. FINAL PUBLISH: Save everything cleanly
    await statelessDb.update(articles)
      .set({
        bodyHtml: finalizedHtml,
        images: images,
        seo_metadata: seoData,
        status: 'PUBLISHED',
        progress_percent: 100,
        current_step_label: "✅ Article Published Successfully",
        validation_logs: [] // Clear logs on success
      })
      .where(eq(articles.id, articleId));

    console.log(`✨ [Job ${articleId}] Successfully published in record time.`);

  } catch (error) {
    console.error(`❌ [Job ${articleId}] Pipeline Failed:`, error.message);
    
    // FAIL GRACEFULLY: Allow the job-recovery cron to pick it up later
    await statelessDb.update(articles)
      .set({
        status: 'FAILED',
        validation_logs: [ `Enterprise Pipeline Error: ${error.message}` ]
      })
      .where(eq(articles.id, articleId));
  }
}
