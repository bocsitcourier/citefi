async draftSection(section, outline, topicData, researchData) {
  this.log('WRITER', `Drafting: ${section.heading}`);

  // STEP 1: Initial Draft
  const initialPrompt = this.buildWriterPrompt(section, outline, topicData, researchData);
  const firstDraft = await this.executeGemini(this.writerModel, initialPrompt);

  // STEP 2: Self-Thinking / Reflexion
  this.log('WRITER', '🧠 Self-Thinking: Reviewing draft for promotional language...');
  const reflexionPrompt = `
    You are an Editor reviewing an article section.
    
    CRITERIA:
    1. Is this section purely educational? (If not conclusion, it MUST be).
    2. Are there promotional "marketing" words (leading, trusted, expert)?
    3. Is the language natural and varied, or does it sound like an AI?
    
    DRAFT: ${firstDraft}
    
    REWRITE the section to be more natural, purely educational, and non-promotional.
  `;
  
  return await this.executeGemini(this.writerModel, reflexionPrompt);
}

Natural Language & AI-Cliche DetectionThe "Smarter" Critic Agent now targets 38 specific AI clichés and marketing "fluff".JavaScript// Enhanced Cliché Detection logic
detectAICliches(text, isConclusion = false) {
  const commonCliches = ['delve into', 'unfold', 'tapestry', 'in today’s world'];
  const promoCliches = ['leading provider', 'our team', 'choose us', 'contact us today'];
  
  let detections = [];
  
  // If not the conclusion, promotional cliches are strictly forbidden
  if (!isConclusion) {
    promoCliches.forEach(cliche => {
      if (text.toLowerCase().includes(cliche)) detections.push({ cliche, type: 'PROMOTIONAL' });
    });
  }
  
  // Natural language check: avoid common AI transitions
  commonCliches.forEach(cliche => {
    if (text.toLowerCase().includes(cliche)) detections.push({ cliche, type: 'AI_CLICHE' });
  });

  return detections;
}
