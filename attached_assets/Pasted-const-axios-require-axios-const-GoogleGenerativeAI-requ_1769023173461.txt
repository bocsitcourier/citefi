const axios = require('axios');
const { GoogleGenerativeAI } = require('@google/generative-ai');

class SmartTopicGenerator {
  constructor(config = {}) {
    this.config = {
      maxTopics: config.maxTopics || 20,
      modelName: 'gemini-2.0-flash-exp', // Fast, huge context, supports thinking
      braveApiKey: config.braveApiKey || process.env.BRAVE_SEARCH_API_KEY,
      geminiApiKey: config.geminiApiKey || process.env.GEMINI_API_KEY,
    };
    
    this.genAI = new GoogleGenerativeAI(this.config.geminiApiKey);
    
    // ANALYST: Precise, deterministic (temp 0.1)
    this.analyst = this.genAI.getGenerativeModel({ 
      model: this.config.modelName, 
      generationConfig: { temperature: 0.1 } 
    });

    // CREATIVE: Catchy, high-variance (temp 0.8)
    this.creative = this.genAI.getGenerativeModel({ 
      model: this.config.modelName, 
      generationConfig: { temperature: 0.8 } 
    });
    
    this.thinkingProcess = [];
  }

  /**
   * THE SMART PIPELINE (Parallel & Agentic)
   */
  async generateTopics(seedTitle, options = {}) {
    this.log(`🚀 Starting Smart Agentic Pipeline for: "${seedTitle}"`);

    try {
      // 1. PARALLEL DISCOVERY: Search + Intent Analysis + Geo Discovery
      // Saves 60% of execution time by running concurrently
      const [searchResults, intentAnalysis, geoEntities] = await Promise.all([
        this.fetchRealWorldData(seedTitle, options),
        this.analyzeSearchIntent(seedTitle, options),
        options.location ? this.discoverLocalEntities(options.location, options.industry) : Promise.resolve([])
      ]);

      // 2. INITIAL GENERATION (The "Draftsman")
      this.log('✍️ Generating initial high-priority drafts...');
      const rawTopics = await this.generateInitialTopics(seedTitle, {
        searchResults, intentAnalysis, geoEntities
      }, options);

      // 3. AGENTIC CRITIQUE (The "Self-Correction" Phase)
      // This identifies hallucinations or generic AI fluff
      const refinedTopics = await this.critiqueAndRefine(rawTopics, searchResults, options);

      // 4. MATHEMATICAL UNIQUENESS (Jaccard Similarity)
      // Score titles based on how different they are from Page 1 of Google
      const finalTopics = refinedTopics.map(topic => {
        const uniqueness = this.calculateUniqueness(topic.title, searchResults.map(r => r.title));
        return { ...topic, uniquenessScore: uniqueness };
      });

      return this.finalizeReport(finalTopics, seedTitle, options);

    } catch (error) {
      this.log(`❌ Critical Error: ${error.message}`);
      throw error;
    }
  }

  /**
   * CRITIQUE LOOP: AI reviews its own work
   */
  async critiqueAndRefine(drafts, searchResults, options) {
    this.log('🧠 Agentic Review: Critiquing titles for hallucinations and cliches...');
    
    const prompt = `
      As a Senior SEO Strategist, review these DRAFT TOPICS for the ${options.industry} industry.
      
      COMPETITOR TITLES ON GOOGLE:
      ${JSON.stringify(searchResults.map(s => s.title))}

      DRAFT TOPICS TO REVIEW:
      ${JSON.stringify(drafts)}

      TASKS:
      1. CRITICAL: Remove any services that are medically/legally impossible for ${options.industry}.
      2. UNIQUENESS: If a title is too similar to the "COMPETITOR TITLES", rewrite it to be a unique angle.
      3. REMOVE CLICHES: Delete anything starting with "The Ultimate Guide to" or "Everything You Need to Know".
      
      Return ONLY a refined JSON array of topic objects.
    `;

    const result = await this.analyst.generateContent(prompt);
    try {
      return JSON.parse(result.response.text().replace(/```json|```/g, ''));
    } catch (e) {
      return drafts; // Fallback to drafts if parsing fails
    }
  }

  /**
   * JACCARD SIMILARITY ENGINE
   * Compares word sets to find mathematical overlap
   */
  calculateUniqueness(newTitle, competitorTitles) {
    const getWords = (str) => new Set(str.toLowerCase().match(/\b(\w+)\b/g));
    const newWords = getWords(newTitle);
    
    let maxSimilarity = 0;

    competitorTitles.forEach(comp => {
      const compWords = getWords(comp);
      const intersection = new Set([...newWords].filter(x => compWords.has(x)));
      const union = new Set([...newWords, ...compWords]);
      const similarity = intersection.size / union.size;
      
      if (similarity > maxSimilarity) maxSimilarity = similarity;
    });

    // Inverse similarity: 100 is perfectly unique, 0 is a duplicate
    return Math.round((1 - maxSimilarity) * 100);
  }

  /**
   * SEARCH DISCOVERY
   */
  async fetchRealWorldData(query, options) {
    this.log('🌐 Fetching real-time search data from Brave...');
    const fullQuery = `${query} ${options.location || ''} ${options.industry || ''}`.trim();
    
    const response = await axios.get('https://api.search.brave.com/res/v1/web/search', {
      params: { q: fullQuery, count: 10 },
      headers: { 'Accept': 'application/json', 'X-Subscription-Token': this.config.braveApiKey }
    });

    return response.data.web.results.map(r => ({
      title: r.title,
      snippet: r.description,
      url: r.url
    }));
  }

  log(message) {
    const timestamp = new Date().toISOString();
    console.log(`[${timestamp}] ${message}`);
    this.thinkingProcess.push({ timestamp, message });
  }

  // ... (Other helper methods: analyzeSearchIntent, discoverLocalEntities, finalizeReport)
}

module.exports = SmartTopicGenerator;
