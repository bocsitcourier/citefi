3. The "Ghost Hunter" (Stuck Job Recovery)You have 18 articles orphaned because of a crash. We need a Self-Healing Cron that uses the statelessDb (Neon HTTP) to rescue these jobs.Add this to lib/job-recovery.ts:TypeScriptimport { statelessDb } from "./db";
import { articles } from "./schema";
import { and, eq, lt, sql } from "drizzle-orm";

export async function rescueOrphanedJobs() {
  const FIFTEEN_MINUTES_AGO = new Date(Date.now() - 15 * 60 * 1000);

  // Find jobs stuck in PROCESSING that haven't been touched in 15 mins
  const orphans = await statelessDb
    .update(articles)
    .set({ 
      status: 'QUEUED', // Reset them to the queue
      validation_logs: sql`array_append(validation_logs, '🛡️ Watchdog: Rescued orphaned job after worker crash')`
    })
    .where(
      and(
        eq(articles.status, 'PROCESSING'),
        lt(articles.updated_at, FIFTEEN_MINUTES_AGO)
      )
    )
    .returning();

  if (orphans.length > 0) {
    console.log(`📡 Watchdog: Rescued ${orphans.length} orphaned jobs.`);
  }
}
