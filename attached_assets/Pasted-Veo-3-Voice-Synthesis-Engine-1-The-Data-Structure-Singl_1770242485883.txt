Veo 3 + Voice Synthesis Engine.1. The Data Structure (Single Source of Truth)Instead of a Markdown script, we define a strict schema. This allows the programmatic pipeline to validate timing before generating expensive media.TypeScript// types/schema.ts
import { z } from 'zod';

export const SceneSchema = z.object({
  id: z.number(),
  timeStart: z.number(),
  duration: z.number(),
  visual: z.object({
    prompt: z.string(),
    cameraMovement: z.enum(['static', 'pan_left', 'pan_right', 'zoom_in', 'truck_in', 'orbit']),
    lighting: z.string(),
    referenceImage: z.string().optional(), // Crucial for Veo 3 character consistency
  }),
  audio: z.object({
    dialogue: z.string(),
    emotion: z.enum(['neutral', 'excited', 'whisper', 'authoritative', 'warm']),
    speaker: z.string(),
    // Instructions for the LLM to generate SSML
    prosodyNotes: z.string(), 
  })
});

export type Scene = z.infer<typeof SceneSchema>;
2. Enhanced Veo 3 Prompt Generator (Multimodal)This version optimizes for Veo 3's specific strengths: Motion Control and Image Conditioning.TypeScript// services/VeoPromptEngine.ts
import { Scene } from '../types/schema';

interface VeoRequestPayload {
  prompt: string;
  negative_prompt?: string;
  image_prompt?: string; // Base64 or URL for character consistency
  video_prompt?: string; // For extending clips
  mask_prompt?: string;  // For inpainting specific areas
  aspect_ratio: string;
  motion_strength: number; // 1-5 scale often used in video models
}

export class VeoPromptEngine {
  private characterRef: Record<string, string>; // Map speaker ID to Ref Image URL

  constructor(characterRefs: Record<string, string>) {
    this.characterRef = characterRefs;
  }

  constructPayload(scene: Scene, previousFrame?: string): VeoRequestPayload {
    // 1. Contextual Construction
    const cameraPrompt = this.formatCamera(scene.visual.cameraMovement);
    
    // 2. Main Prompt Optimization
    const fullPrompt = `
      Cinematic shot, ${scene.visual.lighting}. 
      ${scene.visual.prompt}. 
      Character action: Speaking naturally, subtle head movements.
      ${cameraPrompt}. 
      High fidelity, 8k resolution, raw footage style.
    `.replace(/\s+/g, ' ').trim();

    // 3. Construct Payload
    const payload: VeoRequestPayload = {
      prompt: fullPrompt,
      negative_prompt: "text, watermark, blurry, distorted face, morphing, static image, cartoon",
      aspect_ratio: "16:9",
      motion_strength: scene.visual.cameraMovement === 'static' ? 2 : 4,
    };

    // 4. Handle Continuity (The "Secret Sauce")
    if (previousFrame) {
      // Extending a shot
      payload.video_prompt = previousFrame; 
    } else if (this.characterRef[scene.audio.speaker]) {
      // Starting new shot with consistent character
      payload.image_prompt = this.characterRef[scene.audio.speaker];
    }

    return payload;
  }

  private formatCamera(move: string): string {
    const map: Record<string, string> = {
      zoom_in: "Slow smooth zoom in",
      truck_in: "Camera trucks forward",
      orbit: "Slow orbital camera movement",
      static: "Tripod shot, minimal movement"
    };
    return map[move] || "Handheld organic movement";
  }
}

4. Robust Assembly (Lip-Sync Logic)This updated pipeline uses fluent-ffmpeg and includes logic to handle the "Drift Problem" (where audio is longer than video).TypeScript// services/AssemblyEngine.ts
import ffmpeg from 'fluent-ffmpeg';
import path from 'path';

interface RenderJob {
  videoPath: string;
  audioPath: string;
  outputPath: string;
  duration: number; // The target duration from the script
}

export class AssemblyEngine {
  
  async mergeScene(job: RenderJob): Promise<string> {
    return new Promise((resolve, reject) => {
      // 1. Analyze actual durations
      this.getDuration(job.audioPath).then(audioDur => {
        
        let command = ffmpeg()
          .input(job.videoPath)
          .input(job.audioPath);

        // 2. Intelligent Sync Logic
        // If audio is significantly longer than video (Veo generated too short), 
        // we loop the last frame of video (freeze frame) or slow down video.
        // If video is longer, we trim video.
        
        if (audioDur > job.duration) {
          console.warn(`⚠️ Audio overflow by ${(audioDur - job.duration).toFixed(2)}s. Extending video last frame.`);
          // Complex filter: play video, then hold last frame
          command.complexFilter([
            `[0:v]tpad=stop_mode=clone:stop_duration=${audioDur - job.duration}[v_extended]`,
            `[v_extended][1:a]concat=n=1:v=1:a=1[out]`
          ], ['out']);
        } else {
          // Audio is shorter or equal (Standard)
          // We pad the audio with silence to match video visual timing
          command.complexFilter([
             `[1:a]apad=pad_dur=${job.duration - audioDur}[a_padded]`,
             `[0:v][a_padded]concat=n=1:v=1:a=1[out]`
          ], ['out']);
        }

        command
          .outputOptions('-c:v libx264')
          .outputOptions('-c:a aac')
          .outputOptions('-shortest') // Stop when the shortest stream ends (handled by filters above usually)
          .save(job.outputPath)
          .on('end', () => resolve(job.outputPath))
          .on('error', (err) => reject(err));
      });
    });
  }

  private getDuration(filePath: string): Promise<number> {
    return new Promise((resolve, reject) => {
      ffmpeg.ffprobe(filePath, (err, metadata) => {
        if (err) reject(err);
        else resolve(metadata.format.duration || 0);
      });
    });
  }
}
