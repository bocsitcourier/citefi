This utility handles the three biggest gaps: the Concat List (for stitching), the Shortest-Sync (for audio alignment), and the Ultrafast Preset (to beat the Replit timeout).🧵 The VideoStitcher Utility (lib/video-stitcher.ts)This function takes an array of clip paths and a voiceover path, then builds the single "Master Command" that FFmpeg executes.TypeScriptimport { promises as fs } from 'fs';
import path from 'path';

export async function stitchPlatinumVideo(
  jobId: string, 
  clipPaths: string[], 
  audioPath: string, 
  captionText: string,
  workDir: string
) {
  const listFilePath = path.join(workDir, `concat-${jobId}.txt`);
  const captionFilePath = path.join(workDir, `text-${jobId}.txt`);
  const finalOutput = path.join(workDir, `final-${jobId}.mp4`);
  const FONT_PATH = path.join(process.cwd(), 'assets', 'fonts', 'Inter-Bold.ttf');

  // 1. GAP FIX: Create the Concat List for seamless stitching
  const listContent = clipPaths.map(p => `file '${p}'`).join('\n');
  await fs.writeFile(listFilePath, listContent, 'utf8');

  // 2. GAP FIX: Write captions to file to avoid shell crashes
  await fs.writeFile(captionFilePath, captionText, 'utf8');

  // 3. THE MASTER COMMAND: Stitches, Crops, Captions, and Syncs in ONE pass
  const args = [
    '-hide_banner', '-y',
    '-f', 'concat', '-safe', '0', '-i', listFilePath, // Stitch Clips
    '-i', audioPath,                                  // Add Audio
    '-map', '0:v:0', '-map', '1:a:0',                 // Map streams
    
    // THE FILTER CHAIN: Crop first, then DrawText
    '-vf', `crop=1080:1920:0:0,drawtext=fontfile='${FONT_PATH}':textfile='${captionFilePath}':fontcolor=white:fontsize=44:box=1:boxcolor=black@0.6:boxborderw=20:x=(w-text_w)/2:y=(h-text_h-100)`,
    
    '-c:v', 'libx264',
    '-preset', 'ultrafast', // 🚀 Speed priority
    '-crf', '26',           // File size/Quality balance
    '-c:a', 'aac',          // High compatibility audio
    '-shortest',            // 🚨 CUT audio to match video length exactly
    '-pix_fmt', 'yuv420p',
    finalOutput
  ];

  return { args, finalOutput };
}
