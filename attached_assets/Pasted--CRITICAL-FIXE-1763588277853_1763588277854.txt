// server/index.ts
import { createServer } from 'http';
import { parse } from 'url';
import next from 'next';

const dev = process.env.NODE_ENV !== 'production';
const hostname = 'localhost';
const port = parseInt(process.env.PORT || '3000', 10);

const app = next({ dev, hostname, port });
const handle = app.getRequestHandler();

app.prepare().then(() => {
  createServer(async (req, res) => {
    try {
      const parsedUrl = parse(req.url!, true);
      await handle(req, res, parsedUrl);
    } catch (err) {
      console.error('Error occurred handling', req.url, err);
      res.statusCode = 500;
      res.end('internal server error');
    }
  }).listen(port, () => {
    console.log(`✅ HTTP Server ready on http://${hostname}:${port}`);
  });
});

// server/workers.ts
import PgBoss from 'pg-boss';
import { db } from '@/lib/db';
import { generateArticleWithResearch } from '@/lib/services/article-generator';
import { generateSocialMedia } from '@/lib/services/social-generator';
import { articles, socialPosts } from '@/shared/schema';

const DATABASE_URL = process.env.DATABASE_URL!;

// Initialize pg-boss in worker process
const boss = new PgBoss({
  connectionString: DATABASE_URL,
  max: 10, // Connection pool size
  monitorStateIntervalSeconds: 10,
  newJobCheckIntervalSeconds: 2
});

// ==========================================================================
// ARTICLE GENERATION WORKER (15 CONCURRENT - ALIGNED WITH OPENAI)
// ==========================================================================

boss.work('generate-article', 
  { 
    teamSize: 15, // CRITICAL: Match OpenAI concurrency limit
    teamConcurrency: 1
  },
  async (job: PgBoss.Job<any>) => {
    const startTime = Date.now();
    
    try {
      console.log(`🔄 Processing job ${job.id}: ${job.data.topic}`);

      // Generate article with research
      const result = await generateArticleWithResearch(
        job.data.topic,
        job.data.location || 'United States',
        job.data.keyword || job.data.topic,
        {
          brandData: {
            name: job.data.brandData?.name || 'Your Company',
            industry: job.data.brandData?.industry || 'General'
          }
        }
      );

      // Save to database
      const [article] = await db.insert(articles).values({
        teamId: job.data.teamId,
        userId: job.data.userId,
        batchId: job.data.batchId,
        topic: job.data.topic,
        location: job.data.location,
        keyword: job.data.keyword,
        content: result.article.html,
        wordCount: result.article.wordCount,
        metadata: {
          research: result.research,
          generationTime: ((Date.now() - startTime) / 1000).toFixed(2) + 's'
        }
      }).returning();

      console.log(`✅ Completed job ${job.id} in ${Date.now() - startTime}ms`);

      return {
        success: true,
        articleId: article.id,
        wordCount: article.wordCount
      };

    } catch (error) {
      console.error(`❌ Job ${job.id} failed:`, error);
      
      // Handle specific errors
      if ((error as any).name === 'TimeoutError') {
        throw new Error('Generation timeout - content too complex');
      }
      
      if ((error as any).status === 429) {
        throw new Error('Rate limit exceeded - will retry automatically');
      }

      throw error;
    }
  }
);

// ==========================================================================
// SOCIAL MEDIA WORKER
// ==========================================================================

boss.work('generate-social',
  { teamSize: 10 },
  async (job: PgBoss.Job<any>) => {
    try {
      const article = await db.query.articles.findFirst({
        where: eq(articles.id, job.data.articleId)
      });

      if (!article) {
        throw new Error('Article not found');
      }

      const social = await generateSocialMedia(
        article,
        job.data.platforms || ['linkedin', 'twitter'],
        { tone: job.data.tone }
      );

      // Save social posts
      await Promise.all(
        job.data.platforms.map((platform: string) =>
          db.insert(socialPosts).values({
            teamId: job.data.teamId,
            articleId: article.id,
            platform,
            content: social[platform],
            status: 'draft'
          })
        )
      );

      return { success: true };
    } catch (error) {
      console.error('Social generation failed:', error);
      throw error;
    }
  }
);

// ==========================================================================
// START WORKERS
// ==========================================================================

async function startWorkers() {
  try {
    await boss.start();
    console.log('✅ pg-boss workers started');
    console.log('   - Article generation: 15 concurrent');
    console.log('   - Social media: 10 concurrent');
    
    // Graceful shutdown
    process.on('SIGTERM', async () => {
      console.log('Shutting down workers...');
      await boss.stop();
      process.exit(0);
    });
    
  } catch (error) {
    console.error('Failed to start workers:', error);
    process.exit(1);
  }
}

startWorkers();

// lib/ai/security-utils.ts
import crypto from 'crypto';

/**
 * Constant-time string comparison to prevent timing attacks
 * Critical for API key validation
 */
export function timingSafeEqual(a: string, b: string): boolean {
  if (a.length !== b.length) {
    // Still do a comparison to prevent timing leaks about length
    const dummy = Buffer.from('0'.repeat(32));
    crypto.timingSafeEqual(dummy, dummy);
    return false;
  }

  const bufA = Buffer.from(a);
  const bufB = Buffer.from(b);

  try {
    return crypto.timingSafeEqual(bufA, bufB);
  } catch {
    return false;
  }
}

/**
 * Sanitize error for client response
 * Never expose secrets or internal details in production
 */
export function sanitizeError(error: any, includeDetails = false): {
  message: string;
  code?: string;
  details?: any;
} {
  const isDevelopment = process.env.NODE_ENV === 'development';

  // Remove sensitive information from error messages
  const sanitizedMessage = error.message
    .replace(/sk-[a-zA-Z0-9]{32,}/g, '[REDACTED_KEY]')
    .replace(/Bearer [a-zA-Z0-9_-]+/g, 'Bearer [REDACTED]')
    .replace(/password[=:]\s*\S+/gi, 'password=[REDACTED]');

  return {
    message: sanitizedMessage,
    code: error.code,
    details: (isDevelopment && includeDetails) ? {
      stack: error.stack?.split('\n').slice(0, 3).join('\n'),
      name: error.name
    } : undefined
  };
}

/**
 * Validate API key securely
 */
export function validateAPIKey(provided: string | undefined, expected: string): boolean {
  if (!provided) {
    return false;
  }

  // Remove 'Bearer ' prefix if present
  const cleanProvided = provided.replace(/^Bearer\s+/i, '');
  
  return timingSafeEqual(cleanProvided, expected);
}

/**
 * Secure logger that never logs secrets
 */
export class SecureLogger {
  private static readonly SECRET_PATTERNS = [
    /sk-[a-zA-Z0-9]{32,}/g,
    /Bearer [a-zA-Z0-9_-]+/g,
    /password[=:]\s*\S+/gi,
    /api[_-]?key[=:]\s*\S+/gi
  ];

  static sanitize(message: string): string {
    let sanitized = message;
    for (const pattern of this.SECRET_PATTERNS) {
      sanitized = sanitized.replace(pattern, '[REDACTED]');
    }
    return sanitized;
  }

  static error(message: string, ...args: any[]) {
    console.error(this.sanitize(message), ...args.map(arg => 
      typeof arg === 'string' ? this.sanitize(arg) : arg
    ));
  }

  static info(message: string, ...args: any[]) {
    console.log(this.sanitize(message), ...args);
  }

  static warn(message: string, ...args: any[]) {
    console.warn(this.sanitize(message), ...args);
  }
}

// app/api/health/route.ts
import { NextResponse } from 'next/server';
import { db } from '@/lib/db';
import { sql } from 'drizzle-orm';

export async function GET() {
  const checks: Record<string, any> = {
    timestamp: new Date().toISOString(),
    status: 'healthy',
    checks: {}
  };

  // Check database
  try {
    await db.execute(sql`SELECT 1`);
    checks.checks.database = { status: 'healthy' };
  } catch (error) {
    checks.checks.database = { 
      status: 'unhealthy', 
      error: 'Cannot connect' 
    };
    checks.status = 'unhealthy';
  }

  // Check environment variables
  const requiredEnvVars = [
    'OPENAI_API_KEY',
    'GEMINI_API_KEY',
    'DATABASE_URL'
  ];

  const missingEnvVars = requiredEnvVars.filter(v => !process.env[v]);
  checks.checks.environment = {
    status: missingEnvVars.length === 0 ? 'healthy' : 'unhealthy',
    missing: missingEnvVars.length > 0 ? missingEnvVars : undefined
  };

  if (missingEnvVars.length > 0) {
    checks.status = 'unhealthy';
  }

  // Check OpenAI connectivity
  try {
    const { openaiClient } = await import('@/lib/ai/openai-smart');
    const metrics = openaiClient.getMetrics();
    
    checks.checks.openai = {
      status: 'healthy',
      metrics: {
        totalRequests: metrics.totalRequests,
        successRate: metrics.totalRequests > 0 
          ? ((metrics.successfulRequests / metrics.totalRequests) * 100).toFixed(1) + '%'
          : 'N/A',
        queueSize: metrics.queueSize
      }
    };
  } catch (error) {
    checks.checks.openai = {
      status: 'unknown',
      error: 'Client not initialized'
    };
  }

  const statusCode = checks.status === 'healthy' ? 200 : 503;

  return NextResponse.json(checks, { status: statusCode });
}

export interface GeoEntity {
  name: string;
  address?: string;
  city: string;
  state: string;
  country: string;
  zipCode?: string;
  phone?: string;
  website?: string;
  schema?: any;
}

export class EntityService {
  private entityCache = new Map<string, GeoEntity>();

  /**
   * Get standardized entity data for a location
   * Critical for maintaining entity consistency across content
   */
  async getGeoEntity(location: string, teamId: string): Promise<GeoEntity | null> {
    const cacheKey = `${teamId}:${location}`;

    // Check cache first
    if (this.entityCache.has(cacheKey)) {
      return this.entityCache.get(cacheKey)!;
    }

    // Fetch from database
    try {
      const entity = await db.query.geoEntities.findFirst({
        where: and(
          eq(geoEntities.location, location),
          eq(geoEntities.teamId, teamId)
        )
      });

      if (entity) {
        const geoEntity: GeoEntity = {
          name: entity.name,
          address: entity.address,
          city: entity.city,
          state: entity.state,
          country: entity.country,
          zipCode: entity.zipCode,
          phone: entity.phone,
          website: entity.website,
          schema: entity.schema
        };

        // Cache for 1 hour
        this.entityCache.set(cacheKey, geoEntity);
        setTimeout(() => this.entityCache.delete(cacheKey), 3600000);

        return geoEntity;
      }

      return null;
    } catch (error) {
      console.error('Error fetching geo entity:', error);
      return null;
    }
  }

  /**
   * Inject entity data into article generation prompts
   */
  enrichPromptWithEntity(prompt: string, entity: GeoEntity | null): string {
    if (!entity) {
      return prompt;
    }

    const entityContext = `
VERIFIED LOCAL ENTITY DATA:
Business Name: ${entity.name}
Location: ${entity.city}, ${entity.state}
${entity.address ? `Address: ${entity.address}` : ''}
${entity.phone ? `Phone: ${entity.phone}` : ''}
${entity.website ? `Website: ${entity.website}` : ''}

Use this standardized entity information when referencing local businesses.
`;

    return entityContext + '\n\n' + prompt;
  }

  /**
   * Clear cache (useful for testing)
   */
  clearCache() {
    this.entityCache.clear();
  }
}

// lib/logger.ts
import { SecureLogger } from './ai/security-utils';

export interface LogContext {
  userId?: string;
  teamId?: string;
  jobId?: string;
  articleId?: string;
  [key: string]: any;
}

export class StructuredLogger {
  static log(level: 'info' | 'warn' | 'error', message: string, context?: LogContext) {
    const logEntry = {
      timestamp: new Date().toISOString(),
      level,
      message: SecureLogger.sanitize(message),
      ...context
    };

    const jsonLog = JSON.stringify(logEntry);

    switch (level) {
      case 'error':
        console.error(jsonLog);
        break;
      case 'warn':
        console.warn(jsonLog);
        break;
      default:
        console.log(jsonLog);
    }
  }

  static info(message: string, context?: LogContext) {
    this.log('info', message, context);
  }

  static warn(message: string, context?: LogContext) {
    this.log('warn', message, context);
  }

  static error(message: string, context?: LogContext) {
    this.log('error', message, context);
  }

  /**
   * Log API errors with full context for debugging
   */
  static apiError(
    endpoint: string,
    error: Error,
    context?: LogContext
  ) {
    this.error(`API Error: ${endpoint}`, {
      ...context,
      errorName: error.name,
      errorMessage: error.message,
      stack: process.env.NODE_ENV === 'development' 
        ? error.stack?.split('\n').slice(0, 3)
        : undefined
    });
  }
}

export default StructuredLogger;
