1. The Backend Health API (pages/api/system/health.ts)
This endpoint gathers the raw vitals from your database and the StatusBuffer.

TypeScript
import { db } from '@/lib/db';
import { articles, ai_learning_ledger } from '@/lib/schema';
import { count, eq, sql, and, gte } from 'drizzle-orm';

export default async function handler(req, res) {
  // 1. Get Active Worker Count
  const [active] = await db.select({ val: count() }).from(articles).where(eq(articles.status, 'PROCESSING'));

  // 2. Get Auto-Recoveries (Past 24h)
  const twentyFourHoursAgo = new Date(Date.now() - 24 * 60 * 60 * 1000);
  const [recoveries] = await db.select({ val: count() }).from(ai_learning_ledger)
    .where(and(eq(ai_learning_ledger.category, 'RECOVERY'), gte(ai_learning_ledger.created_at, twentyFourHoursAgo)));

  // 3. Success Rate
  const [published] = await db.select({ val: count() }).from(articles).where(eq(articles.status, 'PUBLISHED'));
  const [failed] = await db.select({ val: count() }).from(articles).where(eq(articles.status, 'FAILED'));

  res.status(200).json({
    workers: active.val,
    recoveries: recoveries.val,
    successRate: Math.round((published.val / (published.val + failed.val || 1)) * 100),
    dbStatus: active.val > 40 ? 'STRESSED' : 'HEALTHY'
  });
}
