It targets text nodes specifically, ensuring attributes and scripts are never touched.TypeScriptimport * as cheerio from 'cheerio';

interface LinkMap {
  [phrase: string]: string;
}

export function applySmartHyperlinks(html: string, linkMap: LinkMap, maxLinks: number = 10): string {
  // 1. Load HTML without adding <html>/<body> wrappers
  const $ = cheerio.load(html, null, false);
  let linksAdded = 0;

  // 2. Create a case-insensitive regex map
  // Sort by length (longest first) to ensure "Boston Nursing Home" matches before "Nursing Home"
  const sortedKeywords = Object.keys(linkMap).sort((a, b) => b.length - a.length);

  // 3. Traverse ONLY text nodes within safe containers
  $('p, li, td, div').each((_, container) => {
    $(container).contents().each((_, node) => {
      if (linksAdded >= maxLinks) return false;

      // Only process actual text nodes
      if (node.type === 'text') {
        let text = $(node).text();
        let modified = false;

        for (const keyword of sortedKeywords) {
          if (linksAdded >= maxLinks) break;

          const regex = new RegExp(`\\b(${escapeRegExp(keyword)})\\b`, 'i');
          const match = text.match(regex);

          if (match) {
            const url = linkMap[keyword];
            const linkedText = `<a href="${url}" class="text-blue-600 underline hover:text-blue-800">${match[0]}</a>`;
            
            // Replace the text node with HTML containing the link
            $(node).replaceWith(text.replace(regex, linkedText));
            linksAdded++;
            modified = true;
            break; // Move to next node to prevent double-linking same text
          }
        }
      }
    });
  });

  return $.html();
}

function escapeRegExp(string: string) {
  return string.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}
