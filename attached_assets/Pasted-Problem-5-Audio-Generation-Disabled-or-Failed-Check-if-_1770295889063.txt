Debug Checklist
Run through this before each generation:
typescriptfunction preGenerationChecklist(scene: SceneSegment): string[] {
  const issues: string[] = [];
  
  // 1. Dialogue format
  if (!scene.dialogue.startsWith('"') && !scene.dialogue.endsWith('"')) {
    // Add quotes if missing
    scene.dialogue = `"${scene.dialogue}"`;
  }
  
  // 2. Speaker specified
  if (!scene.speaker || scene.speaker.trim() === '') {
    issues.push('❌ No speaker specified');
  }
  
  // 3. Dialogue length (Veo struggles with very long dialogue)
  const wordCount = scene.dialogue.split(' ').length;
  if (wordCount > 30) {
    issues.push(`⚠️ Dialogue may be too long (${wordCount} words). Consider splitting.`);
  }
  
  // 4. Duration vs dialogue length
  const estimatedSpeechDuration = wordCount / 2.5; // ~2.5 words per second
  if (estimatedSpeechDuration > scene.duration) {
    issues.push(`⚠️ Dialogue (${estimatedSpeechDuration.toFixed(1)}s) may not fit in ${scene.duration}s scene`);
  }
  
  // 5. Special characters
  if (/[<>{}[\]]/.test(scene.dialogue)) {
    issues.push('❌ Dialogue contains special characters that may cause issues');
  }
  
  // 6. Multiple sentences check
  const sentences = scene.dialogue.split(/[.!?]+/).filter(s => s.trim());
  if (sentences.length > 3) {
    issues.push(`⚠️ ${sentences.length} sentences - consider splitting into multiple scenes`);
  }
  
  return issues;
}

  issues.forEach(i => console.log(`  ${i}`));
