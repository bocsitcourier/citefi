export async function GET(req: Request, { params }: { params: { id: string } }) {
  const batchId = params.id;

  const batchData = await db.query.batches.findFirst({
    where: eq(batches.id, batchId),
    with: {
      articles: {
        // ✅ Only fetch UI-critical columns. 
        // bodyHtml and redditResearch are excluded by default here.
        columns: {
          id: true,
          title: true,
          status: true, 
          progress: true,
          updatedAt: true,
          error: true
        }
      }
    }
  });

  if (!batchData) return new Response("Batch not found", { status: 404 });
  
  return Response.json(batchData);
}

const [pollRate, setPollRate] = useState(3000); // Start at 3 seconds

useEffect(() => {
  if (status !== 'RUNNING') return;

  const fetchUpdate = async () => {
    try {
      await refetch();
      // 📈 Slowly increase the delay between polls (max 15s)
      // This prevents UI-induced DB starvation during long runs
      setPollRate(prev => Math.min(prev + 1000, 15000));
    } catch (err) {
      console.error("Polling error:", err);
    }
  };

  const timer = setTimeout(fetchUpdate, pollRate);
  
  return () => clearTimeout(timer);
}, [status, pollRate, refetch]);
